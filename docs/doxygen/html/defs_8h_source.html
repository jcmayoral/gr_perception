51  return x.format == format && x.width == width && x.height == height;
52  }
53  /*
54  When width = 0, height = 0, minDimension = 0, and maxDimension = 0,
55  keep the orignal frame resolution
56  When width = 0, height = 0, minDimension != 0, and maxDimension = 0,
57  keep the aspect ratio and resize the frame so that shorter edge size is
58  minDimension
59  When width = 0, height = 0, minDimension = 0, and maxDimension != 0,
60  keep the aspect ratio and resize the frame so that longer edge size is
61  maxDimension
62  When width = 0, height = 0, minDimension != 0, and maxDimension != 0,
63  resize the frame so that shorter edge size is minDimension, and
64  longer edge size is maxDimension. The aspect ratio may not be preserved
65  When width = 0, height != 0, minDimension = 0, and maxDimension = 0,
66  keep the aspect ratio and resize the frame so that frame height is $height
67  When width != 0, height = 0, minDimension = 0, and maxDimension = 0,
68  keep the aspect ratio and resize the frame so that frame width is $width
69  When width != 0, height != 0, minDimension = 0, and maxDimension = 0,
70  resize the frame so that frame width and height are set to $width and
71  $height,
72  respectively
73  */
74  size_t width{0}; // width in pixels
75  size_t height{0}; // height in pixels
76  long format{-1}; // AVPixelFormat, auto AV_PIX_FMT_NONE
77  size_t minDimension{0}; // choose min dimension and rescale accordingly

249 /*
250  * External provider of the ecnoded bytes, specific implementation is left for
251  * different use cases, like file, memory, external network end-points, etc.
252  * Normally input/output parameter @out set to valid, not null buffer pointer,
253  * which indicates "read" call, however there are "seek" modes as well.
254 
255  * @out != nullptr => read from the current offset, @whence got ignored,
256  * @size bytes to read => return number bytes got read, 0 if no more bytes
257  * available, < 0 on error.
258 
259  * @out == nullptr, @timeoutMs == 0 => does provider support "seek"
260  * capability in a first place? @size & @whence got ignored, return 0 on
261  * success, < 0 if "seek" mode is not supported.
262 
263  * @out == nullptr, @timeoutMs != 0 => normal seek call
264  * offset == @size, i.e. @whence = [SEEK_SET, SEEK_CUR, SEEK_END, AVSEEK_SIZE)
265  * return < 0 on error, position if @whence = [SEEK_SET, SEEK_CUR, SEEK_END],
266  * length of buffer if @whence = [AVSEEK_SIZE].
267  */
