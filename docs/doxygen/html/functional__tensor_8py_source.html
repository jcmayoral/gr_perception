128 def adjust_hue(img, hue_factor):
129  """Adjust hue of an image.
130 
131  The image hue is adjusted by converting the image to HSV and
132  cyclically shifting the intensities in the hue channel (H).
133  The image is then converted back to original image mode.
134 
135  `hue_factor` is the amount of shift in H channel and must be in the
136  interval `[-0.5, 0.5]`.
137 
138  See `Hue`_ for more details.
139 
140  .. _Hue: https://en.wikipedia.org/wiki/Hue
141 
142  Args:
143  img (Tensor): Image to be adjusted. Image type is either uint8 or float.
144  hue_factor (float): How much to shift the hue channel. Should be in
145  [-0.5, 0.5]. 0.5 and -0.5 give complete reversal of hue channel in
146  HSV space in positive and negative direction respectively.
147  0 means no shift. Therefore, both -0.5 and 0.5 will give an image
148  with complementary colors while 0 gives the original image.
149 
150  Returns:
151  Tensor: Hue adjusted image.
152  """
153  if not (-0.5 <= hue_factor <= 0.5):
154  raise ValueError('hue_factor ({}) is not in [-0.5, 0.5].'.format(hue_factor))
155 
156  if not _is_tensor_a_torch_image(img):
157  raise TypeError('tensor is not a torch image.')
158 
159  orig_dtype = img.dtype
160  if img.dtype == torch.uint8:
161  img = img.to(dtype=torch.float32) / 255.0
162 
163  img = _rgb2hsv(img)
164  h, s, v = img.unbind(0)
165  h += hue_factor
166  h = h % 1.0
167  img = torch.stack((h, s, v))
168  img_hue_adj = _hsv2rgb(img)
169 
170  if orig_dtype == torch.uint8:
171  img_hue_adj = (img_hue_adj * 255.0).to(dtype=orig_dtype)
172 
173  return img_hue_adj

302 def _rgb2hsv(img):
303  r, g, b = img.unbind(0)
304 
305  maxc = torch.max(img, dim=0).values
306  minc = torch.min(img, dim=0).values
307 
308  # The algorithm erases S and H channel where `maxc = minc`. This avoids NaN
309  # from happening in the results, because
310  # + S channel has division by `maxc`, which is zero only if `maxc = minc`
311  # + H channel has division by `(maxc - minc)`.
312  #
313  # Instead of overwriting NaN afterwards, we just prevent it from occuring so
314  # we don't need to deal with it in case we save the NaN in a buffer in
315  # backprop, if it is ever supported, but it doesn't hurt to do so.
316  eqc = maxc == minc
317 
318  cr = maxc - minc
319  # Since `eqc => cr = 0`, replacing denominator with 1 when `eqc` is fine.
320  s = cr / torch.where(eqc, maxc.new_ones(()), maxc)
321  # Note that `eqc => maxc = minc = r = g = b`. So the following calculation
322  # of `h` would reduce to `bc - gc + 2 + rc - bc + 4 + rc - bc = 6` so it
323  # would not matter what values `rc`, `gc`, and `bc` have here, and thus
324  # replacing denominator with 1 when `eqc` is fine.
325  cr_divisor = torch.where(eqc, maxc.new_ones(()), cr)
326  rc = (maxc - r) / cr_divisor
327  gc = (maxc - g) / cr_divisor
328  bc = (maxc - b) / cr_divisor
329 
330  hr = (maxc == r) * (bc - gc)
331  hg = ((maxc == g) & (maxc != r)) * (2.0 + rc - bc)
332  hb = ((maxc != g) & (maxc != r)) * (4.0 + gc - rc)
333  h = (hr + hg + hb)
334  h = torch.fmod((h / 6.0 + 1.0), 1.0)
335  return torch.stack((h, s, maxc))

338 def _hsv2rgb(img):
339  h, s, v = img.unbind(0)
340  i = torch.floor(h * 6.0)
341  f = (h * 6.0) - i
342  i = i.to(dtype=torch.int32)
343 
344  p = torch.clamp((v * (1.0 - s)), 0.0, 1.0)
345  q = torch.clamp((v * (1.0 - s * f)), 0.0, 1.0)
346  t = torch.clamp((v * (1.0 - s * (1.0 - f))), 0.0, 1.0)
347  i = i % 6
348 
349  mask = i == torch.arange(6)[:, None, None]
350 
351  a1 = torch.stack((v, q, p, p, t, v))
352  a2 = torch.stack((t, v, v, q, p, p))
353  a3 = torch.stack((p, p, t, v, v, q))
354  a4 = torch.stack((a1, a2, a3))
355 
356  return torch.einsum("ijk, xijk -> xjk", mask.to(dtype=img.dtype), a4)

359 def _pad_symmetric(img: Tensor, padding: List[int]) -> Tensor:
360  # padding is left, right, top, bottom
361  in_sizes = img.size()
362 
363  x_indices = [i for i in range(in_sizes[-1])] # [0, 1, 2, 3, ...]
364  left_indices = [i for i in range(padding[0] - 1, -1, -1)] # e.g. [3, 2, 1, 0]
365  right_indices = [-(i + 1) for i in range(padding[1])] # e.g. [-1, -2, -3]
366  x_indices = torch.tensor(left_indices + x_indices + right_indices)
367 
368  y_indices = [i for i in range(in_sizes[-2])]
369  top_indices = [i for i in range(padding[2] - 1, -1, -1)]
370  bottom_indices = [-(i + 1) for i in range(padding[3])]
371  y_indices = torch.tensor(top_indices + y_indices + bottom_indices)
372 
373  ndim = img.ndim
374  if ndim == 3:
375  return img[:, y_indices[:, None], x_indices[None, :]]
376  elif ndim == 4:
377  return img[:, :, y_indices[:, None], x_indices[None, :]]
378  else:
