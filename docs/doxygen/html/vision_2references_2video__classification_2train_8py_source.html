24 def train_one_epoch(model, criterion, optimizer, lr_scheduler, data_loader, device, epoch, print_freq, apex=False):
25  model.train()
26  metric_logger = utils.MetricLogger(delimiter=" ")
27  metric_logger.add_meter('lr', utils.SmoothedValue(window_size=1, fmt='{value}'))
28  metric_logger.add_meter('clips/s', utils.SmoothedValue(window_size=10, fmt='{value:.3f}'))
29 
30  header = 'Epoch: [{}]'.format(epoch)
31  for video, target in metric_logger.log_every(data_loader, print_freq, header):
32  start_time = time.time()
33  video, target = video.to(device), target.to(device)
34  output = model(video)
35  loss = criterion(output, target)
36 
37  optimizer.zero_grad()
38  if apex:
39  with amp.scale_loss(loss, optimizer) as scaled_loss:
40  scaled_loss.backward()
41  else:
42  loss.backward()
43  optimizer.step()
44 
45  acc1, acc5 = utils.accuracy(output, target, topk=(1, 5))
46  batch_size = video.shape[0]
47  metric_logger.update(loss=loss.item(), lr=optimizer.param_groups[0]["lr"])
48  metric_logger.meters['acc1'].update(acc1.item(), n=batch_size)
49  metric_logger.meters['acc5'].update(acc5.item(), n=batch_size)
50  metric_logger.meters['clips/s'].update(batch_size / (time.time() - start_time))
51  lr_scheduler.step()
