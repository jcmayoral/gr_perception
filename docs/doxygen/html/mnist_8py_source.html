443 def read_sn3_pascalvincent_tensor(path, strict=True):
444  """Read a SN3 file in "Pascal Vincent" format (Lush file 'libidx/idx-io.lsh').
445  Argument may be a filename, compressed filename, or file object.
446  """
447  # typemap
448  if not hasattr(read_sn3_pascalvincent_tensor, 'typemap'):
449  read_sn3_pascalvincent_tensor.typemap = {
450  8: (torch.uint8, np.uint8, np.uint8),
451  9: (torch.int8, np.int8, np.int8),
452  11: (torch.int16, np.dtype('>i2'), 'i2'),
453  12: (torch.int32, np.dtype('>i4'), 'i4'),
454  13: (torch.float32, np.dtype('>f4'), 'f4'),
455  14: (torch.float64, np.dtype('>f8'), 'f8')}
456  # read
457  with open_maybe_compressed_file(path) as f:
458  data = f.read()
459  # parse
460  magic = get_int(data[0:4])
461  nd = magic % 256
462  ty = magic // 256
463  assert nd >= 1 and nd <= 3
464  assert ty >= 8 and ty <= 14
465  m = read_sn3_pascalvincent_tensor.typemap[ty]
466  s = [get_int(data[4 * (i + 1): 4 * (i + 2)]) for i in range(nd)]
467  parsed = np.frombuffer(data, dtype=m[1], offset=(4 * (nd + 1)))
468  assert parsed.shape[0] == np.prod(s) or not strict
469  return torch.from_numpy(parsed.astype(m[2], copy=False)).view(*s)
