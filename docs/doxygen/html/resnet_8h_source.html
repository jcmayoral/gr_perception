93 template <typename Block>
94 torch::nn::Sequential ResNetImpl<Block>::_make_layer(
95  int64_t planes,
96  int64_t blocks,
97  int64_t stride) {
98  torch::nn::Sequential downsample = nullptr;
99  if (stride != 1 || inplanes != planes * Block::expansion) {
100  downsample = torch::nn::Sequential(
101  _resnetimpl::conv1x1(inplanes, planes * Block::expansion, stride),
102  torch::nn::BatchNorm2d(planes * Block::expansion));
103  }
104 
105  torch::nn::Sequential layers;
106  layers->push_back(
107  Block(inplanes, planes, stride, downsample, groups, base_width));
108 
109  inplanes = planes * Block::expansion;
110 
111  for (int i = 1; i < blocks; ++i)
112  layers->push_back(Block(inplanes, planes, 1, nullptr, groups, base_width));
113 
114  return layers;
115 }

118 ResNetImpl<Block>::ResNetImpl(
119  const std::vector<int>& layers,
120  int64_t num_classes,
121  bool zero_init_residual,
122  int64_t groups,

129  bn1(64),
130  layer1(_make_layer(64, layers[0])),
131  layer2(_make_layer(128, layers[1], 2)),
132  layer3(_make_layer(256, layers[2], 2)),
133  layer4(_make_layer(512, layers[3], 2)),
134  fc(512 * Block::expansion, num_classes) {
135  register_module("conv1", conv1);
136  register_module("bn1", bn1);
137  register_module("fc", fc);
138 
139  register_module("layer1", layer1);
140  register_module("layer2", layer2);
141  register_module("layer3", layer3);
142  register_module("layer4", layer4);
143 
144  for (auto& module : modules(/*include_self=*/false)) {
145  if (auto M = dynamic_cast<torch::nn::Conv2dImpl*>(module.get()))
146  torch::nn::init::kaiming_normal_(
147  M->weight,
148  /*a=*/0,
149  torch::kFanOut,
150  torch::kReLU);
151  else if (auto M = dynamic_cast<torch::nn::BatchNorm2dImpl*>(module.get())) {
152  torch::nn::init::constant_(M->weight, 1);
153  torch::nn::init::constant_(M->bias, 0);
154  }
155  }

157  // Zero-initialize the last BN in each residual branch, so that the residual
158  // branch starts with zeros, and each residual block behaves like an
159  // identity. This improves the model by 0.2~0.3% according to
160  // https://arxiv.org/abs/1706.02677
161  if (zero_init_residual)
162  for (auto& module : modules(/*include_self=*/false)) {
163  if (auto* M = dynamic_cast<_resnetimpl::Bottleneck*>(module.get()))
164  torch::nn::init::constant_(M->bn3->weight, 0);
165  else if (auto* M = dynamic_cast<_resnetimpl::BasicBlock*>(module.get()))
166  torch::nn::init::constant_(M->bn2->weight, 0);
167  }
168 }

171 torch::Tensor ResNetImpl<Block>::forward(torch::Tensor x) {
172  x = conv1->forward(x);
173  x = bn1->forward(x).relu_();
174  x = torch::max_pool2d(x, 3, 2, 1);
175 
176  x = layer1->forward(x);
177  x = layer2->forward(x);
178  x = layer3->forward(x);
179  x = layer4->forward(x);
180 
181  x = torch::adaptive_avg_pool2d(x, {1, 1});
182  x = x.reshape({x.size(0), -1});
183  x = fc->forward(x);
184 
185  return x;
186 }
