26  pairwise_dist = torch.cdist(embeddings, embeddings, p=p)
27 
28  mask_anchor_positive = _get_anchor_positive_triplet_mask(labels).float()
29  anchor_positive_dist = mask_anchor_positive * pairwise_dist
30 
31  # hardest positive for every anchor
32  hardest_positive_dist, _ = anchor_positive_dist.max(1, keepdim=True)
33 
34  mask_anchor_negative = _get_anchor_negative_triplet_mask(labels).float()
35 
36  # Add max value in each row to invalid negatives
37  max_anchor_negative_dist, _ = pairwise_dist.max(1, keepdim=True)
38  anchor_negative_dist = pairwise_dist + max_anchor_negative_dist * (1.0 - mask_anchor_negative)
39 
40  # hardest negative for every anchor
41  hardest_negative_dist, _ = anchor_negative_dist.min(1, keepdim=True)
42 
43  triplet_loss = hardest_positive_dist - hardest_negative_dist + margin
44  triplet_loss[triplet_loss < 0] = 0

62  # Remove negative losses (easy triplets)
63  triplet_loss[triplet_loss < 0] = 0
64

71

73  triplet_loss = triplet_loss.sum() / (num_positive_triplets + 1e-16)
