27 def inception_v3(pretrained=False, progress=True, quantize=False, **kwargs):
28  r"""Inception v3 model architecture from
29  `"Rethinking the Inception Architecture for Computer Vision" <http://arxiv.org/abs/1512.00567>`_.
30 
31  .. note::
32  **Important**: In contrast to the other models the inception_v3 expects tensors with a size of
33  N x 3 x 299 x 299, so ensure your images are sized accordingly.
34 
35  Note that quantize = True returns a quantized model with 8 bit
36  weights. Quantized models only support inference and run on CPUs.
37  GPU inference is not yet supported
38 
39  Args:
40  pretrained (bool): If True, returns a model pre-trained on ImageNet
41  progress (bool): If True, displays a progress bar of the download to stderr
42  aux_logits (bool): If True, add an auxiliary branch that can improve training.
43  Default: *True*
44  transform_input (bool): If True, preprocesses the input according to the method with which it
45  was trained on ImageNet. Default: *False*
46  """
47  if pretrained:
48  if "transform_input" not in kwargs:
49  kwargs["transform_input"] = True
50  if "aux_logits" in kwargs:
51  original_aux_logits = kwargs["aux_logits"]
52  kwargs["aux_logits"] = True
53  else:
54  original_aux_logits = False
55 
56  model = QuantizableInception3(**kwargs)
57  _replace_relu(model)
58 
59  if quantize:
60  # TODO use pretrained as a string to specify the backend
61  backend = 'fbgemm'
62  quantize_model(model, backend)
63  else:
64  assert pretrained in [True, False]
65 
66  if pretrained:
67  if quantize:
68  if not original_aux_logits:
69  model.aux_logits = False
70  del model.AuxLogits
71  model_url = quant_model_urls['inception_v3_google' + '_' + backend]
72  else:
73  model_url = inception_module.model_urls['inception_v3_google']
74 
75  state_dict = load_state_dict_from_url(model_url,
76  progress=progress)
77 
78  model.load_state_dict(state_dict)
79 
80  if not quantize:
81  if not original_aux_logits:
82  model.aux_logits = False
83  del model.AuxLogits
