128 def convert_image_dtype(image: torch.Tensor, dtype: torch.dtype = torch.float) -> torch.Tensor:
129  """Convert a tensor image to the given ``dtype`` and scale the values accordingly
130 
131  Args:
132  image (torch.Tensor): Image to be converted
133  dtype (torch.dtype): Desired data type of the output
134 
135  Returns:
136  (torch.Tensor): Converted image
137 
138  .. note::
139 
140  When converting from a smaller to a larger integer ``dtype`` the maximum values are **not** mapped exactly.
141  If converted back and forth, this mismatch has no effect.
142 
143  Raises:
144  RuntimeError: When trying to cast :class:`torch.float32` to :class:`torch.int32` or :class:`torch.int64` as
145  well as for trying to cast :class:`torch.float64` to :class:`torch.int64`. These conversions might lead to
146  overflow errors since the floating point ``dtype`` cannot store consecutive integers over the whole range
147  of the integer ``dtype``.
148  """
149  if image.dtype == dtype:
150  return image
151 
152  if image.dtype.is_floating_point:
153  # float to float
154  if dtype.is_floating_point:
155  return image.to(dtype)
156 
157  # float to int
158  if (image.dtype == torch.float32 and dtype in (torch.int32, torch.int64)) or (
159  image.dtype == torch.float64 and dtype == torch.int64
160  ):
161  msg = f"The cast from {image.dtype} to {dtype} cannot be performed safely."
162  raise RuntimeError(msg)
163 
164  eps = 1e-3
165  return image.mul(torch.iinfo(dtype).max + 1 - eps).to(dtype)
166  else:
167  # int to float
168  if dtype.is_floating_point:
169  max = torch.iinfo(image.dtype).max
170  image = image.to(dtype)
171  return image / max
172 
173  # int to int
174  input_max = torch.iinfo(image.dtype).max
175  output_max = torch.iinfo(dtype).max
176 
177  if input_max > output_max:
178  factor = (input_max + 1) // (output_max + 1)
179  image = image // factor
180  return image.to(dtype)
181  else:
182  factor = (output_max + 1) // (input_max + 1)
183  image = image.to(dtype)
184  return image * factor

187 def to_pil_image(pic, mode=None):
188  """Convert a tensor or an ndarray to PIL Image.
189 
190  See :class:`~torchvision.transforms.ToPILImage` for more details.
191 
192  Args:
193  pic (Tensor or numpy.ndarray): Image to be converted to PIL Image.
194  mode (`PIL.Image mode`_): color space and pixel depth of input data (optional).
195 
196  .. _PIL.Image mode: https://pillow.readthedocs.io/en/latest/handbook/concepts.html#concept-modes
197 
198  Returns:
199  PIL Image: Image converted to PIL Image.
200  """
201  if not(isinstance(pic, torch.Tensor) or isinstance(pic, np.ndarray)):
202  raise TypeError('pic should be Tensor or ndarray. Got {}.'.format(type(pic)))
203 
204  elif isinstance(pic, torch.Tensor):
205  if pic.ndimension() not in {2, 3}:
206  raise ValueError('pic should be 2/3 dimensional. Got {} dimensions.'.format(pic.ndimension()))
207 
208  elif pic.ndimension() == 2:
209  # if 2D image, add channel dimension (CHW)
210  pic = pic.unsqueeze(0)
211 
212  elif isinstance(pic, np.ndarray):
213  if pic.ndim not in {2, 3}:
214  raise ValueError('pic should be 2/3 dimensional. Got {} dimensions.'.format(pic.ndim))
215 
216  elif pic.ndim == 2:
217  # if 2D image, add channel dimension (HWC)
218  pic = np.expand_dims(pic, 2)
219 
220  npimg = pic
221  if isinstance(pic, torch.FloatTensor) and mode != 'F':
222  pic = pic.mul(255).byte()
223  if isinstance(pic, torch.Tensor):
224  npimg = np.transpose(pic.numpy(), (1, 2, 0))
225 
226  if not isinstance(npimg, np.ndarray):
227  raise TypeError('Input pic must be a torch.Tensor or NumPy ndarray, ' +
228  'not {}'.format(type(npimg)))
229 
230  if npimg.shape[2] == 1:
231  expected_mode = None
232  npimg = npimg[:, :, 0]
233  if npimg.dtype == np.uint8:
234  expected_mode = 'L'
235  elif npimg.dtype == np.int16:
236  expected_mode = 'I;16'
237  elif npimg.dtype == np.int32:
238  expected_mode = 'I'
239  elif npimg.dtype == np.float32:
240  expected_mode = 'F'
241  if mode is not None and mode != expected_mode:
242  raise ValueError("Incorrect mode ({}) supplied for input type {}. Should be {}"
243  .format(mode, np.dtype, expected_mode))
244  mode = expected_mode
245 
246  elif npimg.shape[2] == 2:
247  permitted_2_channel_modes = ['LA']
248  if mode is not None and mode not in permitted_2_channel_modes:
249  raise ValueError("Only modes {} are supported for 2D inputs".format(permitted_2_channel_modes))
250 
251  if mode is None and npimg.dtype == np.uint8:
252  mode = 'LA'
253 
254  elif npimg.shape[2] == 4:
255  permitted_4_channel_modes = ['RGBA', 'CMYK', 'RGBX']
256  if mode is not None and mode not in permitted_4_channel_modes:
257  raise ValueError("Only modes {} are supported for 4D inputs".format(permitted_4_channel_modes))
258 
259  if mode is None and npimg.dtype == np.uint8:
260  mode = 'RGBA'
261  else:
262  permitted_3_channel_modes = ['RGB', 'YCbCr', 'HSV']
263  if mode is not None and mode not in permitted_3_channel_modes:
264  raise ValueError("Only modes {} are supported for 3D inputs".format(permitted_3_channel_modes))
265  if mode is None and npimg.dtype == np.uint8:
266  mode = 'RGB'
267 
268  if mode is None:
269  raise TypeError('Input type {} is not supported'.format(npimg.dtype))
270 
271  return Image.fromarray(npimg, mode=mode)

274 def normalize(tensor, mean, std, inplace=False):
275  """Normalize a tensor image with mean and standard deviation.
276 
277  .. note::
278  This transform acts out of place by default, i.e., it does not mutates the input tensor.
279 
280  See :class:`~torchvision.transforms.Normalize` for more details.
281 
282  Args:
283  tensor (Tensor): Tensor image of size (C, H, W) to be normalized.
284  mean (sequence): Sequence of means for each channel.
285  std (sequence): Sequence of standard deviations for each channel.
286  inplace(bool,optional): Bool to make this operation inplace.
287 
288  Returns:
289  Tensor: Normalized Tensor image.
290  """
291  if not torch.is_tensor(tensor):
292  raise TypeError('tensor should be a torch tensor. Got {}.'.format(type(tensor)))
293 
294  if tensor.ndimension() != 3:
295  raise ValueError('Expected tensor to be a tensor image of size (C, H, W). Got tensor.size() = '
296  '{}.'.format(tensor.size()))
297 
298  if not inplace:
299  tensor = tensor.clone()
300 
301  dtype = tensor.dtype
302  mean = torch.as_tensor(mean, dtype=dtype, device=tensor.device)
303  std = torch.as_tensor(std, dtype=dtype, device=tensor.device)
304  if (std == 0).any():
305  raise ValueError('std evaluated to zero after conversion to {}, leading to division by zero.'.format(dtype))
306  if mean.ndim == 1:
307  mean = mean[:, None, None]
308  if std.ndim == 1:
309  std = std[:, None, None]
310  tensor.sub_(mean).div_(std)
311  return tensor

347 def pad(img: Tensor, padding: List[int], fill: int = 0, padding_mode: str = "constant") -> Tensor:
348  r"""Pad the given image on all sides with the given "pad" value.
349  The image can be a PIL Image or a torch Tensor, in which case it is expected
350  to have [..., H, W] shape, where ... means an arbitrary number of leading dimensions
351 
352  Args:
353  img (PIL Image or Tensor): Image to be padded.
354  padding (int or tuple or list): Padding on each border. If a single int is provided this
355  is used to pad all borders. If tuple of length 2 is provided this is the padding
356  on left/right and top/bottom respectively. If a tuple of length 4 is provided
357  this is the padding for the left, top, right and bottom borders respectively.
358  In torchscript mode padding as single int is not supported, use a tuple or
359  list of length 1: ``[padding, ]``.
360  fill (int or str or tuple): Pixel fill value for constant fill. Default is 0. If a tuple of
361  length 3, it is used to fill R, G, B channels respectively.
362  This value is only used when the padding_mode is constant. Only int value is supported for Tensors.
363  padding_mode: Type of padding. Should be: constant, edge, reflect or symmetric. Default is constant.
364  Mode symmetric is not yet supported for Tensor inputs.
365 
366  - constant: pads with a constant value, this value is specified with fill
367 
368  - edge: pads with the last value on the edge of the image
369 
370  - reflect: pads with reflection of image (without repeating the last value on the edge)
371 
372  padding [1, 2, 3, 4] with 2 elements on both sides in reflect mode
373  will result in [3, 2, 1, 2, 3, 4, 3, 2]
374 
375  - symmetric: pads with reflection of image (repeating the last value on the edge)
376 
377  padding [1, 2, 3, 4] with 2 elements on both sides in symmetric mode
378  will result in [2, 1, 1, 2, 3, 4, 4, 3]
379 
380  Returns:
381  PIL Image or Tensor: Padded image.
382  """
383  if not isinstance(img, torch.Tensor):
384  return F_pil.pad(img, padding=padding, fill=fill, padding_mode=padding_mode)

489 def _parse_fill(fill, img, min_pil_version):
490  """Helper function to get the fill color for rotate and perspective transforms.
491 
492  Args:
493  fill (n-tuple or int or float): Pixel fill value for area outside the transformed
494  image. If int or float, the value is used for all bands respectively.
495  Defaults to 0 for all bands.
496  img (PIL Image): Image to be filled.
497  min_pil_version (str): The minimum PILLOW version for when the ``fillcolor`` option
498  was first introduced in the calling function. (e.g. rotate->5.2.0, perspective->5.0.0)
499 
500  Returns:
501  dict: kwarg for ``fillcolor``
502  """
503  major_found, minor_found = (int(v) for v in PILLOW_VERSION.split('.')[:2])
504  major_required, minor_required = (int(v) for v in min_pil_version.split('.')[:2])
505  if major_found < major_required or (major_found == major_required and minor_found < minor_required):
506  if fill is None:
507  return {}
508  else:
509  msg = ("The option to fill background area of the transformed image, "
510  "requires pillow>={}")
511  raise RuntimeError(msg.format(min_pil_version))
512 
513  num_bands = len(img.getbands())
514  if fill is None:
515  fill = 0
516  if isinstance(fill, (int, float)) and num_bands > 1:
517  fill = tuple([fill] * num_bands)
518  if not isinstance(fill, (int, float)) and len(fill) != num_bands:
519  msg = ("The number of elements in 'fill' does not match the number of "
520  "bands of the image ({} != {})")
521  raise ValueError(msg.format(len(fill), num_bands))
522 
523  return {"fillcolor": fill}

526 def _get_perspective_coeffs(startpoints, endpoints):
527  """Helper function to get the coefficients (a, b, c, d, e, f, g, h) for the perspective transforms.
528 
529  In Perspective Transform each pixel (x, y) in the original image gets transformed as,
530  (x, y) -> ( (ax + by + c) / (gx + hy + 1), (dx + ey + f) / (gx + hy + 1) )
531 
532  Args:
533  List containing [top-left, top-right, bottom-right, bottom-left] of the original image,
534  List containing [top-left, top-right, bottom-right, bottom-left] of the transformed image
535  Returns:
536  octuple (a, b, c, d, e, f, g, h) for transforming each pixel.
537  """
538  matrix = []
539 
540  for p1, p2 in zip(endpoints, startpoints):
541  matrix.append([p1[0], p1[1], 1, 0, 0, 0, -p2[0] * p1[0], -p2[0] * p1[1]])
542  matrix.append([0, 0, 0, p1[0], p1[1], 1, -p2[1] * p1[0], -p2[1] * p1[1]])
543 
544  A = torch.tensor(matrix, dtype=torch.float)
545  B = torch.tensor(startpoints, dtype=torch.float).view(8)
546  res = torch.lstsq(B, A)[0]
547  return res.squeeze_(1).tolist()

594  """Crop the given image into four corners and the central crop.
595  The image can be a PIL Image or a Tensor, in which case it is expected
596  to have [..., H, W] shape, where ... means an arbitrary number of leading dimensions

607

615  size = (size[0], size[0])

618  raise ValueError("Please provide only two dimensions (h, w) for size.")

621  crop_height, crop_width = size
622  if crop_width > image_width or crop_height > image_height:
623  msg = "Requested crop size {} is bigger than input size {}"

763 def adjust_gamma(img, gamma, gain=1):
764  r"""Perform gamma correction on an image.
765 
766  Also known as Power Law Transform. Intensities in RGB mode are adjusted
767  based on the following equation:
768 
769  .. math::
770  I_{\text{out}} = 255 \times \text{gain} \times \left(\frac{I_{\text{in}}}{255}\right)^{\gamma}

772  See `Gamma Correction`_ for more details.

775 
776  Args:
777  img (PIL Image): PIL Image to be adjusted.
778  gamma (float): Non negative real number, same as :math:`\gamma` in the equation.
779  gamma larger than 1 make the shadows darker,
780  while gamma smaller than 1 make dark regions lighter.
781  gain (float): The constant multiplier.
782  """
783  if not F_pil._is_pil_image(img):
784  raise TypeError('img should be PIL Image. Got {}'.format(type(img)))
785 
786  if gamma < 0:
787  raise ValueError('Gamma should be a non-negative real number')
788 
789  input_mode = img.mode
790  img = img.convert('RGB')
791 
792  gamma_map = [255 * gain * pow(ele / 255., gamma) for ele in range(256)] * 3
793  img = img.point(gamma_map) # use PIL's point-function to accelerate this part
794 
795  img = img.convert(input_mode)
796  return img

831 def _get_inverse_affine_matrix(center, angle, translate, scale, shear):
832  # Helper method to compute inverse matrix for affine transformation
833 
834  # As it is explained in PIL.Image.rotate
835  # We need compute INVERSE of affine transformation matrix: M = T * C * RSS * C^-1
836  # where T is translation matrix: [1, 0, tx | 0, 1, ty | 0, 0, 1]
837  # C is translation matrix to keep center: [1, 0, cx | 0, 1, cy | 0, 0, 1]
838  # RSS is rotation with scale and shear matrix
839  # RSS(a, s, (sx, sy)) =
840  # = R(a) * S(s) * SHy(sy) * SHx(sx)
841  # = [ s*cos(a - sy)/cos(sy), s*(-cos(a - sy)*tan(x)/cos(y) - sin(a)), 0 ]
842  # [ s*sin(a + sy)/cos(sy), s*(-sin(a - sy)*tan(x)/cos(y) + cos(a)), 0 ]
843  # [ 0 , 0 , 1 ]
844  #
845  # where R is a rotation matrix, S is a scaling matrix, and SHx and SHy are the shears:
846  # SHx(s) = [1, -tan(s)] and SHy(s) = [1 , 0]
847  # [0, 1 ] [-tan(s), 1]
848  #
849  # Thus, the inverse is M^-1 = C * RSS^-1 * C^-1 * T^-1
850 
851  if isinstance(shear, numbers.Number):
852  shear = [shear, 0]
853 
854  if not isinstance(shear, (tuple, list)) and len(shear) == 2:
855  raise ValueError(
856  "Shear should be a single value or a tuple/list containing " +
857  "two values. Got {}".format(shear))
858 
859  rot = math.radians(angle)
860  sx, sy = [math.radians(s) for s in shear]
861 
862  cx, cy = center
863  tx, ty = translate
864 
865  # RSS without scaling
866  a = cos(rot - sy) / cos(sy)
867  b = -cos(rot - sy) * tan(sx) / cos(sy) - sin(rot)
868  c = sin(rot - sy) / cos(sy)
869  d = -sin(rot - sy) * tan(sx) / cos(sy) + cos(rot)
870 
871  # Inverted rotation matrix with scale and shear
872  # det([[a, b], [c, d]]) == 1, since det(rotation) = 1 and det(shear) = 1
873  M = [d, -b, 0,
874  -c, a, 0]
875  M = [x / scale for x in M]
876 
877  # Apply inverse of translation and of center translation: RSS^-1 * C^-1 * T^-1
878  M[2] += M[0] * (-cx - tx) + M[1] * (-cy - ty)
879  M[5] += M[3] * (-cx - tx) + M[4] * (-cy - ty)
880 
881  # Apply center translation: C * RSS^-1 * C^-1 * T^-1
882  M[2] += cx
883  M[5] += cy
884  return M
