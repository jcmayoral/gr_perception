13 # copying result_idx_in_level to a specific index in result[]
14 # is not supported by ONNX tracing yet.
15 # _onnx_merge_levels() is an implementation supported by ONNX
16 # that merges the levels to the right indices
17 @torch.jit.unused
18 def _onnx_merge_levels(levels: Tensor, unmerged_results: List[Tensor]) -> Tensor:
19  first_result = unmerged_results[0]
20  dtype, device = first_result.dtype, first_result.device
21  res = torch.zeros((levels.size(0), first_result.size(1),
22  first_result.size(2), first_result.size(3)),
23  dtype=dtype, device=device)
24  for level in range(len(unmerged_results)):
25  index = (levels == level).nonzero().view(-1, 1, 1, 1)
26  index = index.expand(index.size(0),
27  unmerged_results[level].size(1),
28  unmerged_results[level].size(2),
29  unmerged_results[level].size(3))
30  res = res.scatter(0, index, unmerged_results[level])
31  return res

45 class LevelMapper(object):
46  """Determine which FPN level each RoI in a set of RoIs should map to based
47  on the heuristic in the FPN paper.
48 
49  Arguments:
50  k_min (int)
51  k_max (int)
52  canonical_scale (int)
53  canonical_level (int)
54  eps (float)
55  """
56 
57  def __init__(
58  self,
59  k_min: int,
60  k_max: int,
61  canonical_scale: int = 224,
62  canonical_level: int = 4,
63  eps: float = 1e-6,
64  ):
65  self.k_min = k_min
66  self.k_max = k_max
67  self.s0 = canonical_scale
68  self.lvl0 = canonical_level
69  self.eps = eps
70 
71  def __call__(self, boxlists: List[Tensor]) -> Tensor:
72  """
73  Arguments:
74  boxlists (list[BoxList])
75  """
76  # Compute level ids
77  s = torch.sqrt(torch.cat([box_area(boxlist) for boxlist in boxlists]))
78 
79  # Eqn.(1) in FPN paper
80  target_lvls = torch.floor(self.lvl0 + torch.log2(s / self.s0) + torch.tensor(self.eps, dtype=s.dtype))
81  target_lvls = torch.clamp(target_lvls, min=self.k_min, max=self.k_max)
82  return (target_lvls.to(torch.int64) - self.k_min).to(torch.int64)

85 class MultiScaleRoIAlign(nn.Module):
86  """
87  Multi-scale RoIAlign pooling, which is useful for detection with or without FPN.
88 
89  It infers the scale of the pooling via the heuristics present in the FPN paper.
90 
91  Arguments:
92  featmap_names (List[str]): the names of the feature maps that will be used
93  for the pooling.
94  output_size (List[Tuple[int, int]] or List[int]): output size for the pooled region
95  sampling_ratio (int): sampling ratio for ROIAlign
96 
97  Examples::
98 
99  >>> m = torchvision.ops.MultiScaleRoIAlign(['feat1', 'feat3'], 3, 2)
100  >>> i = OrderedDict()
101  >>> i['feat1'] = torch.rand(1, 5, 64, 64)
102  >>> i['feat2'] = torch.rand(1, 5, 32, 32) # this feature won't be used in the pooling
103  >>> i['feat3'] = torch.rand(1, 5, 16, 16)
104  >>> # create some random bounding boxes
105  >>> boxes = torch.rand(6, 4) * 256; boxes[:, 2:] += boxes[:, :2]
106  >>> # original image size, before computing the feature maps
107  >>> image_sizes = [(512, 512)]
108  >>> output = m(i, [boxes], image_sizes)
109  >>> print(output.shape)
110  >>> torch.Size([6, 5, 3, 3])
111 
112  """
113

115  'scales': Optional[List[float]],
116  'map_levels': Optional[LevelMapper]
117  }

134  def convert_to_roi_format(self, boxes: List[Tensor]) -> Tensor:
135  concat_boxes = torch.cat(boxes, dim=0)
136  device, dtype = concat_boxes.device, concat_boxes.dtype
137  ids = torch.cat(
138  [
139  torch.full_like(b[:, :1], i, dtype=dtype, layout=torch.strided, device=device)
140  for i, b in enumerate(boxes)
141  ],
142  dim=0,
143  )
144  rois = torch.cat([ids, concat_boxes], dim=1)
145  return rois

147  def infer_scale(self, feature: Tensor, original_size: List[int]) -> float:
148  # assumption: the scale is of the form 2 ** (-k), with k integer
149  size = feature.shape[-2:]
150  possible_scales = torch.jit.annotate(List[float], [])
151  for s1, s2 in zip(size, original_size):
152  approx_scale = float(s1) / float(s2)
153  scale = 2 ** float(torch.tensor(approx_scale).log2().round())
154  possible_scales.append(scale)
155  assert possible_scales[0] == possible_scales[1]
156  return possible_scales[0]

158  def setup_scales(
159  self,
160  features: List[Tensor],
161  image_shapes: List[Tuple[int, int]],
162  ) -> None:
163  assert len(image_shapes) != 0
164  max_x = 0
165  max_y = 0
166  for shape in image_shapes:
167  max_x = max(shape[0], max_x)
168  max_y = max(shape[1], max_y)
169  original_input_shape = (max_x, max_y)
170 
171  scales = [self.infer_scale(feat, original_input_shape) for feat in features]
172  # get the levels in the feature map by leveraging the fact that the network always
173  # downsamples by a factor of 2 at each level.
174  lvl_min = -torch.log2(torch.tensor(scales[0], dtype=torch.float32)).item()
175  lvl_max = -torch.log2(torch.tensor(scales[-1], dtype=torch.float32)).item()
176  self.scales = scales
177  self.map_levels = initLevelMapper(int(lvl_min), int(lvl_max))

185  """
186  Arguments:
187  x (OrderedDict[Tensor]): feature maps for each level. They are assumed to have
188  all the same number of channels, but they can have different sizes.

192  image_shapes (List[Tuple[height, width]]): the sizes of each image before they

195  Returns:
196  result (Tensor)

242

245  else:
246  result[idx_in_level] = result_idx_in_level
247 
248  if torchvision._is_tracing():
249  result = _onnx_merge_levels(levels, tracing_results)
