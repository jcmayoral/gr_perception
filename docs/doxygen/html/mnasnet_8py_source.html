23 class _InvertedResidual(nn.Module):
24 
25  def __init__(self, in_ch, out_ch, kernel_size, stride, expansion_factor,
26  bn_momentum=0.1):
27  super(_InvertedResidual, self).__init__()
28  assert stride in [1, 2]
29  assert kernel_size in [3, 5]
30  mid_ch = in_ch * expansion_factor
31  self.apply_residual = (in_ch == out_ch and stride == 1)
32  self.layers = nn.Sequential(
33  # Pointwise
34  nn.Conv2d(in_ch, mid_ch, 1, bias=False),
35  nn.BatchNorm2d(mid_ch, momentum=bn_momentum),
36  nn.ReLU(inplace=True),
37  # Depthwise
38  nn.Conv2d(mid_ch, mid_ch, kernel_size, padding=kernel_size // 2,
39  stride=stride, groups=mid_ch, bias=False),
40  nn.BatchNorm2d(mid_ch, momentum=bn_momentum),
41  nn.ReLU(inplace=True),
42  # Linear pointwise. Note that there's no activation.
43  nn.Conv2d(mid_ch, out_ch, 1, bias=False),
44  nn.BatchNorm2d(out_ch, momentum=bn_momentum))
45 
46  def forward(self, input):
47  if self.apply_residual:
48  return self.layers(input) + input
49  else:
50  return self.layers(input)

53 def _stack(in_ch, out_ch, kernel_size, stride, exp_factor, repeats,
54  bn_momentum):
55  """ Creates a stack of inverted residuals. """
56  assert repeats >= 1
57  # First one has no skip, because feature map size changes.
58  first = _InvertedResidual(in_ch, out_ch, kernel_size, stride, exp_factor,
59  bn_momentum=bn_momentum)
60  remaining = []
61  for _ in range(1, repeats):
62  remaining.append(
63  _InvertedResidual(out_ch, out_ch, kernel_size, 1, exp_factor,
64  bn_momentum=bn_momentum))
65  return nn.Sequential(first, *remaining)

68 def _round_to_multiple_of(val, divisor, round_up_bias=0.9):
69  """ Asymmetric rounding to make `val` divisible by `divisor`. With default
70  bias, will round up, unless the number is no more than 10% greater than the
71  smaller divisible value, i.e. (83, 8) -> 80, but (84, 8) -> 88. """
72  assert 0.0 < round_up_bias < 1.0
73  new_val = max(divisor, int(val + divisor / 2) // divisor * divisor)
74  return new_val if new_val >= round_up_bias * val else new_val + divisor

77 def _get_depths(alpha):
78  """ Scales tensor depths as in reference MobileNet code, prefers rouding up
79  rather than down. """
80  depths = [32, 16, 24, 40, 80, 96, 192, 320]
81  return [_round_to_multiple_of(depth * alpha, 8) for depth in depths]

84 class MNASNet(torch.nn.Module):
85  """ MNASNet, as described in https://arxiv.org/pdf/1807.11626.pdf. This
86  implements the B1 variant of the model.
87  >>> model = MNASNet(1000, 1.0)
88  >>> x = torch.rand(1, 3, 224, 224)
89  >>> y = model(x)
90  >>> y.dim()
91  1
92  >>> y.nelement()
93  1000
94  """
95  # Version 2 adds depth scaling in the initial stages of the network.
96  _version = 2
97 
98  def __init__(self, alpha, num_classes=1000, dropout=0.2):
99  super(MNASNet, self).__init__()
100  assert alpha > 0.0
101  self.alpha = alpha
102  self.num_classes = num_classes
103  depths = _get_depths(alpha)
104  layers = [
105  # First layer: regular conv.
106  nn.Conv2d(3, depths[0], 3, padding=1, stride=2, bias=False),
107  nn.BatchNorm2d(depths[0], momentum=_BN_MOMENTUM),
108  nn.ReLU(inplace=True),
109  # Depthwise separable, no skip.
110  nn.Conv2d(depths[0], depths[0], 3, padding=1, stride=1,
111  groups=depths[0], bias=False),
112  nn.BatchNorm2d(depths[0], momentum=_BN_MOMENTUM),
113  nn.ReLU(inplace=True),
114  nn.Conv2d(depths[0], depths[1], 1, padding=0, stride=1, bias=False),
115  nn.BatchNorm2d(depths[1], momentum=_BN_MOMENTUM),
116  # MNASNet blocks: stacks of inverted residuals.
117  _stack(depths[1], depths[2], 3, 2, 3, 3, _BN_MOMENTUM),
118  _stack(depths[2], depths[3], 5, 2, 3, 3, _BN_MOMENTUM),
119  _stack(depths[3], depths[4], 5, 2, 6, 3, _BN_MOMENTUM),
120  _stack(depths[4], depths[5], 3, 1, 6, 2, _BN_MOMENTUM),
121  _stack(depths[5], depths[6], 5, 2, 6, 4, _BN_MOMENTUM),
122  _stack(depths[6], depths[7], 3, 1, 6, 1, _BN_MOMENTUM),
123  # Final mapping to classifier input.
124  nn.Conv2d(depths[7], 1280, 1, padding=0, stride=1, bias=False),
125  nn.BatchNorm2d(1280, momentum=_BN_MOMENTUM),
126  nn.ReLU(inplace=True),
127  ]
128  self.layers = nn.Sequential(*layers)
129  self.classifier = nn.Sequential(nn.Dropout(p=dropout, inplace=True),
130  nn.Linear(1280, num_classes))
131  self._initialize_weights()
132 
133  def forward(self, x):
134  x = self.layers(x)
135  # Equivalent to global avgpool and removing H and W dimensions.
136  x = x.mean([2, 3])
137  return self.classifier(x)

139  def _initialize_weights(self):
140  for m in self.modules():
141  if isinstance(m, nn.Conv2d):
142  nn.init.kaiming_normal_(m.weight, mode="fan_out",
143  nonlinearity="relu")
144  if m.bias is not None:
145  nn.init.zeros_(m.bias)
146  elif isinstance(m, nn.BatchNorm2d):
147  nn.init.ones_(m.weight)
148  nn.init.zeros_(m.bias)
149  elif isinstance(m, nn.Linear):
150  nn.init.kaiming_uniform_(m.weight, mode="fan_out",
151  nonlinearity="sigmoid")
152  nn.init.zeros_(m.bias)

154  def _load_from_state_dict(self, state_dict, prefix, local_metadata, strict,
155  missing_keys, unexpected_keys, error_msgs):
156  version = local_metadata.get("version", None)
157  assert version in [1, 2]
158 
159  if version == 1 and not self.alpha == 1.0:
160  # In the initial version of the model (v1), stem was fixed-size.
161  # All other layer configurations were the same. This will patch
162  # the model so that it's identical to v1. Model with alpha 1.0 is
163  # unaffected.
164  depths = _get_depths(self.alpha)
165  v1_stem = [
166  nn.Conv2d(3, 32, 3, padding=1, stride=2, bias=False),
167  nn.BatchNorm2d(32, momentum=_BN_MOMENTUM),
168  nn.ReLU(inplace=True),
169  nn.Conv2d(32, 32, 3, padding=1, stride=1, groups=32,
170  bias=False),
171  nn.BatchNorm2d(32, momentum=_BN_MOMENTUM),
172  nn.ReLU(inplace=True),
173  nn.Conv2d(32, 16, 1, padding=0, stride=1, bias=False),
174  nn.BatchNorm2d(16, momentum=_BN_MOMENTUM),
175  _stack(16, depths[2], 3, 2, 3, 3, _BN_MOMENTUM),
176  ]
177  for idx, layer in enumerate(v1_stem):
178  self.layers[idx] = layer
179 
180  # The model is now identical to v1, and must be saved as such.
181  self._version = 1
182  warnings.warn(
183  "A new version of MNASNet model has been implemented. "
184  "Your checkpoint was saved using the previous version. "
185  "This checkpoint will load and work as before, but "
186  "you may want to upgrade by training a newer model or "
187  "transfer learning from an updated ImageNet checkpoint.",
188  UserWarning)

195 def _load_pretrained(model_name, model, progress):
196  if model_name not in _MODEL_URLS or _MODEL_URLS[model_name] is None:
197  raise ValueError(
198  "No checkpoint is available for model type {}".format(model_name))
199  checkpoint_url = _MODEL_URLS[model_name]
