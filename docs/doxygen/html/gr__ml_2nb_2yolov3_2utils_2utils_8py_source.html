36 def rescale_boxes(boxes, current_dim, original_shape):
37  """ Rescales bounding boxes to the original shape """
38  orig_h, orig_w = original_shape
39  # The amount of padding that was added
40  pad_x = max(orig_h - orig_w, 0) * (current_dim / max(original_shape))
41  pad_y = max(orig_w - orig_h, 0) * (current_dim / max(original_shape))
42  # Image height and width after padding is removed
43  unpad_h = current_dim - pad_y
44  unpad_w = current_dim - pad_x
45  # Rescale bounding boxes to dimension of original image
46  boxes[:, 0] = ((boxes[:, 0] - pad_x // 2) / unpad_w) * orig_w
47  boxes[:, 1] = ((boxes[:, 1] - pad_y // 2) / unpad_h) * orig_h
48  boxes[:, 2] = ((boxes[:, 2] - pad_x // 2) / unpad_w) * orig_w
49  boxes[:, 3] = ((boxes[:, 3] - pad_y // 2) / unpad_h) * orig_h
50  return boxes

71 def ap_per_class(tp, conf, pred_cls, target_cls):
72  """ Compute the average precision, given the recall and precision curves.
73  Source: https://github.com/rafaelpadilla/Object-Detection-Metrics.
74  # Arguments
75  tp: True positives (list).
76  conf: Objectness value from 0-1 (list).
77  pred_cls: Predicted object classes (list).
78  target_cls: True object classes (list).
79  # Returns
80  The average precision as computed in py-faster-rcnn.
81  """
82 
83  # Sort by objectness
84  i = np.argsort(-conf)
85  tp, conf, pred_cls = tp[i], conf[i], pred_cls[i]
86 
87  # Find unique classes
88  unique_classes = np.unique(target_cls)
89 
90  # Create Precision-Recall curve and compute AP for each class
91  ap, p, r = [], [], []
92  for c in tqdm.tqdm(unique_classes, desc="Computing AP"):
93  i = pred_cls == c
94  n_gt = (target_cls == c).sum() # Number of ground truth objects
95  n_p = i.sum() # Number of predicted objects
96 
97  if n_p == 0 and n_gt == 0:
98  continue
99  elif n_p == 0 or n_gt == 0:
100  ap.append(0)
101  r.append(0)
102  p.append(0)
103  else:
104  # Accumulate FPs and TPs
105  fpc = (1 - tp[i]).cumsum()
106  tpc = (tp[i]).cumsum()
107 
108  # Recall
109  recall_curve = tpc / (n_gt + 1e-16)
110  r.append(recall_curve[-1])
111 
112  # Precision
113  precision_curve = tpc / (tpc + fpc)
114  p.append(precision_curve[-1])
115 
116  # AP from recall-precision curve
117  ap.append(compute_ap(recall_curve, precision_curve))
118 
119  # Compute F1 score (harmonic mean of precision and recall)
120  p, r, ap = np.array(p), np.array(r), np.array(ap)
121  f1 = 2 * p * r / (p + r + 1e-16)
122 
123  return p, r, ap, f1, unique_classes.astype("int32")

126 def compute_ap(recall, precision):
127  """ Compute the average precision, given the recall and precision curves.
128  Code originally from https://github.com/rbgirshick/py-faster-rcnn.
129 
130  # Arguments
131  recall: The recall curve (list).
132  precision: The precision curve (list).
133  # Returns
134  The average precision as computed in py-faster-rcnn.
135  """
136  # correct AP calculation
137  # first append sentinel values at the end
138  mrec = np.concatenate(([0.0], recall, [1.0]))
139  mpre = np.concatenate(([0.0], precision, [0.0]))
140 
141  # compute the precision envelope
142  for i in range(mpre.size - 1, 0, -1):
143  mpre[i - 1] = np.maximum(mpre[i - 1], mpre[i])
144 
145  # to calculate area under PR curve, look for points
146  # where X axis (recall) changes value
147  i = np.where(mrec[1:] != mrec[:-1])[0]
148 
149  # and sum (\Delta recall) * prec
150  ap = np.sum((mrec[i + 1] - mrec[i]) * mpre[i + 1])
151  return ap

202 def bbox_iou(box1, box2, x1y1x2y2=True):
203  """
204  Returns the IoU of two bounding boxes
205  """
206  if not x1y1x2y2:
207  # Transform from center and width to exact coordinates
208  b1_x1, b1_x2 = box1[:, 0] - box1[:, 2] / 2, box1[:, 0] + box1[:, 2] / 2
209  b1_y1, b1_y2 = box1[:, 1] - box1[:, 3] / 2, box1[:, 1] + box1[:, 3] / 2
210  b2_x1, b2_x2 = box2[:, 0] - box2[:, 2] / 2, box2[:, 0] + box2[:, 2] / 2
211  b2_y1, b2_y2 = box2[:, 1] - box2[:, 3] / 2, box2[:, 1] + box2[:, 3] / 2
212  else:
213  # Get the coordinates of bounding boxes
214  b1_x1, b1_y1, b1_x2, b1_y2 = box1[:, 0], box1[:, 1], box1[:, 2], box1[:, 3]
215  b2_x1, b2_y1, b2_x2, b2_y2 = box2[:, 0], box2[:, 1], box2[:, 2], box2[:, 3]
216 
217  # get the corrdinates of the intersection rectangle
218  inter_rect_x1 = torch.max(b1_x1, b2_x1)
219  inter_rect_y1 = torch.max(b1_y1, b2_y1)
220  inter_rect_x2 = torch.min(b1_x2, b2_x2)
221  inter_rect_y2 = torch.min(b1_y2, b2_y2)
222  # Intersection area
223  inter_area = torch.clamp(inter_rect_x2 - inter_rect_x1 + 1, min=0) * torch.clamp(
224  inter_rect_y2 - inter_rect_y1 + 1, min=0
225  )
226  # Union Area
227  b1_area = (b1_x2 - b1_x1 + 1) * (b1_y2 - b1_y1 + 1)
228  b2_area = (b2_x2 - b2_x1 + 1) * (b2_y2 - b2_y1 + 1)
229 
230  iou = inter_area / (b1_area + b2_area - inter_area + 1e-16)
231 
232  return iou

235 def non_max_suppression(prediction, conf_thres=0.5, nms_thres=0.4):
236  """
237  Removes detections with lower object confidence score than 'conf_thres' and performs
238  Non-Maximum Suppression to further filter detections.
239  Returns detections with shape:
240  (x1, y1, x2, y2, object_conf, class_score, class_pred)
241  """
242 
243  # From (center x, center y, width, height) to (x1, y1, x2, y2)
244  prediction[..., :4] = xywh2xyxy(prediction[..., :4])
245  output = [None for _ in range(len(prediction))]
246  for image_i, image_pred in enumerate(prediction):
247  # Filter out confidence scores below threshold
248  image_pred = image_pred[image_pred[:, 4] >= conf_thres]
249  # If none are remaining => process next image
250  if not image_pred.size(0):
251  continue
252  # Object confidence times class confidence
253  score = image_pred[:, 4] * image_pred[:, 5:].max(1)[0]
254  # Sort by it
255  image_pred = image_pred[(-score).argsort()]
256  class_confs, class_preds = image_pred[:, 5:].max(1, keepdim=True)
257  detections = torch.cat((image_pred[:, :5], class_confs.float(), class_preds.float()), 1)
258  # Perform non-maximum suppression
259  keep_boxes = []
260  while detections.size(0):
261  large_overlap = bbox_iou(detections[0, :4].unsqueeze(0), detections[:, :4]) > nms_thres
262  label_match = detections[0, -1] == detections[:, -1]
263  # Indices of boxes with lower confidence scores, large IOUs and matching labels
264  invalid = large_overlap & label_match
265  weights = detections[invalid, 4:5]
266  # Merge overlapping bboxes by order of confidence
267  detections[0, :4] = (weights * detections[invalid, :4]).sum(0) / weights.sum()
268  keep_boxes += [detections[0]]
269  detections = detections[~invalid]
270  if keep_boxes:
271  output[image_i] = torch.stack(keep_boxes)
272 
273  return output

276 def build_targets(pred_boxes, pred_cls, target, anchors, ignore_thres):

296 
297  # Convert to position relative to box
298  target_boxes = target[:, 2:6] * nG
299  gxy = target_boxes[:, :2]

301  # Get anchors with best iou
302  ious = torch.stack([bbox_wh_iou(anchor, gwh) for anchor in anchors])
303  best_ious, best_n = ious.max(0)

314  for i, anchor_ious in enumerate(ious.t()):
315  noobj_mask[b[i], anchor_ious > ignore_thres, gj[i], gi[i]] = 0
316

318  tx[b, best_n, gj, gi] = gx - gx.floor()
319  ty[b, best_n, gj, gi] = gy - gy.floor()
320  # Width and height
321  tw[b, best_n, gj, gi] = torch.log(gw / anchors[best_n][:, 0] + 1e-16)
322  th[b, best_n, gj, gi] = torch.log(gh / anchors[best_n][:, 1] + 1e-16)
323  # One-hot encoding of label
324  tcls[b, best_n, gj, gi, target_labels] = 1
325  # Compute label correctness and iou at best anchor
326  class_mask[b, best_n, gj, gi] = (pred_cls[b, best_n, gj, gi].argmax(-1) == target_labels).float()
327  iou_scores[b, best_n, gj, gi] = bbox_iou(pred_boxes[b, best_n, gj, gi], target_boxes, x1y1x2y2=False)

329  tconf = obj_mask.float()
330  return iou_scores, class_mask, obj_mask, noobj_mask, tx, ty, tw, th, tcls, tconf
