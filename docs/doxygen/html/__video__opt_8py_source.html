127 def _align_audio_frames(aframes, aframe_pts, audio_pts_range):
128  # type: (torch.Tensor, torch.Tensor, List[int]) -> torch.Tensor
129  start, end = aframe_pts[0], aframe_pts[-1]
130  num_samples = aframes.size(0)
131  step_per_aframe = float(end - start + 1) / float(num_samples)
132  s_idx = 0
133  e_idx = num_samples
134  if start < audio_pts_range[0]:
135  s_idx = int((audio_pts_range[0] - start) / step_per_aframe)
136  if end > audio_pts_range[1]:
137  e_idx = int((audio_pts_range[1] - end) / step_per_aframe)
138  return aframes[s_idx:e_idx, :]

182  edge size is video_min_dimension, and longer edge size is
183  video_max_dimension. The aspect ratio may not be preserved
184  - When video_width = 0, video_height != 0, video_min_dimension = 0,
185  and video_max_dimension = 0, keep the aspect ratio and resize
186  the frame so that frame video_height is $video_height
187  - When video_width != 0, video_height == 0, video_min_dimension = 0,
188  and video_max_dimension = 0, keep the aspect ratio and resize
189  the frame so that frame video_width is $video_width
190  - When video_width != 0, video_height != 0, video_min_dimension = 0,
191  and video_max_dimension = 0, resize the frame so that frame
192  video_width and video_height are set to $video_width and
193  $video_height, respectively
194  video_pts_range : list(int), optional
195  the start and end presentation timestamp of video stream
196  video_timebase: Fraction, optional

200  audio_samples: int, optional

474  if end_pts is None:

477  if pts_unit == "pts":

488  def get_pts(time_base):
489  start_offset = start_pts
490  end_offset = end_pts
491  if pts_unit == "sec":
492  start_offset = int(math.floor(start_pts * (1 / time_base)))
493  if end_offset != float("inf"):
494  end_offset = int(math.ceil(end_pts * (1 / time_base)))
495  if end_offset == float("inf"):
496  end_offset = -1
497  return start_offset, end_offset
498 
499  video_pts_range = (0, -1)
500  video_timebase = default_timebase
501  if has_video:

504  )

507  audio_pts_range = (0, -1)
508  audio_timebase = default_timebase

511  info.audio_timebase.numerator, info.audio_timebase.denominator
512  )
513  audio_pts_range = get_pts(audio_timebase)
