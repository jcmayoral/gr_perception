33  """
34  Module that generates anchors for a set of feature maps and
35  image sizes.
36 
37  The module support computing anchors at multiple sizes and aspect ratios
38  per feature map. This module assumes aspect ratio = height / width for
39  each anchor.
40 
41  sizes and aspect_ratios should have the same number of elements, and it should
42  correspond to the number of feature maps.
43 
44  sizes[i] and aspect_ratios[i] can have an arbitrary number of elements,
45  and AnchorGenerator will output a set of sizes[i] * aspect_ratios[i] anchors
46  per spatial location for feature map i.
47 
48  Arguments:
49  sizes (Tuple[Tuple[int]]):
50  aspect_ratios (Tuple[Tuple[float]]):
51  """
52 
53  def __init__(
54  self,
55  sizes=(128, 256, 512),
56  aspect_ratios=(0.5, 1.0, 2.0),
57  ):
58  super(AnchorGenerator, self).__init__()
59 
60  if not isinstance(sizes[0], (list, tuple)):
61  # TODO change this
62  sizes = tuple((s,) for s in sizes)
63  if not isinstance(aspect_ratios[0], (list, tuple)):
64  aspect_ratios = (aspect_ratios,) * len(sizes)
65 
66  assert len(sizes) == len(aspect_ratios)
67 
68  self.sizes = sizes
69  self.aspect_ratios = aspect_ratios
70  self.cell_anchors = None
71  self._cache = {}
72 
73  # TODO: https://github.com/pytorch/pytorch/issues/26792
74  # For every (aspect_ratios, scales) combination, output a zero-centered anchor with those values.
75  # (scales, aspect_ratios) are usually an element of zip(self.scales, self.aspect_ratios)
76  # This method assumes aspect ratio = height / width for an anchor.
77  def generate_anchors(self, scales, aspect_ratios, dtype=torch.float32, device="cpu"):
78  # type: (List[int], List[float], int, Device) -> Tensor # noqa: F821
79  scales = torch.as_tensor(scales, dtype=dtype, device=device)
80  aspect_ratios = torch.as_tensor(aspect_ratios, dtype=dtype, device=device)
81  h_ratios = torch.sqrt(aspect_ratios)
82  w_ratios = 1 / h_ratios
83 
84  ws = (w_ratios[:, None] * scales[None, :]).view(-1)
85  hs = (h_ratios[:, None] * scales[None, :]).view(-1)
86 
87  base_anchors = torch.stack([-ws, -hs, ws, hs], dim=1) / 2
88  return base_anchors.round()
89 
90  def set_cell_anchors(self, dtype, device):
91  # type: (int, Device) -> None # noqa: F821
92  if self.cell_anchors is not None:
93  cell_anchors = self.cell_anchors
94  assert cell_anchors is not None
95  # suppose that all anchors have the same device
96  # which is a valid assumption in the current state of the codebase
97  if cell_anchors[0].device == device:
98  return
99 
100  cell_anchors = [
101  self.generate_anchors(
102  sizes,
103  aspect_ratios,
104  dtype,
105  device
106  )
107  for sizes, aspect_ratios in zip(self.sizes, self.aspect_ratios)
108  ]
109  self.cell_anchors = cell_anchors
110 
111  def num_anchors_per_location(self):
112  return [len(s) * len(a) for s, a in zip(self.sizes, self.aspect_ratios)]
113 
114  # For every combination of (a, (g, s), i) in (self.cell_anchors, zip(grid_sizes, strides), 0:2),
115  # output g[i] anchors that are s[i] distance apart in direction i, with the same dimensions as a.
116  def grid_anchors(self, grid_sizes, strides):
117  # type: (List[List[int]], List[List[Tensor]]) -> List[Tensor]
118  anchors = []
119  cell_anchors = self.cell_anchors
120  assert cell_anchors is not None
121 
122  for size, stride, base_anchors in zip(
123  grid_sizes, strides, cell_anchors
124  ):
125  grid_height, grid_width = size
126  stride_height, stride_width = stride
127  device = base_anchors.device
128 
129  # For output anchor, compute [x_center, y_center, x_center, y_center]
130  shifts_x = torch.arange(
131  0, grid_width, dtype=torch.float32, device=device
132  ) * stride_width
133  shifts_y = torch.arange(
134  0, grid_height, dtype=torch.float32, device=device
135  ) * stride_height
136  shift_y, shift_x = torch.meshgrid(shifts_y, shifts_x)
137  shift_x = shift_x.reshape(-1)
138  shift_y = shift_y.reshape(-1)
139  shifts = torch.stack((shift_x, shift_y, shift_x, shift_y), dim=1)
140 
141  # For every (base anchor, output anchor) pair,
142  # offset each zero-centered base anchor by the center of the output anchor.
143  anchors.append(
144  (shifts.view(-1, 1, 4) + base_anchors.view(1, -1, 4)).reshape(-1, 4)
145  )
146 
147  return anchors

149  def cached_grid_anchors(self, grid_sizes, strides):
150  # type: (List[List[int]], List[List[Tensor]]) -> List[Tensor]
151  key = str(grid_sizes) + str(strides)
152  if key in self._cache:
153  return self._cache[key]
154  anchors = self.grid_anchors(grid_sizes, strides)
155  self._cache[key] = anchors
156  return anchors

158  def forward(self, image_list, feature_maps):
159  # type: (ImageList, List[Tensor]) -> List[Tensor]
160  grid_sizes = list([feature_map.shape[-2:] for feature_map in feature_maps])
161  image_size = image_list.tensors.shape[-2:]
162  dtype, device = feature_maps[0].dtype, feature_maps[0].device
163  strides = [[torch.tensor(image_size[0] // g[0], dtype=torch.int64, device=device),
164  torch.tensor(image_size[1] // g[1], dtype=torch.int64, device=device)] for g in grid_sizes]
165  self.set_cell_anchors(dtype, device)
166  anchors_over_all_feature_maps = self.cached_grid_anchors(grid_sizes, strides)
167  anchors = torch.jit.annotate(List[List[torch.Tensor]], [])
168  for i, (image_height, image_width) in enumerate(image_list.image_sizes):
169  anchors_in_image = []
170  for anchors_per_feature_map in anchors_over_all_feature_maps:
171  anchors_in_image.append(anchors_per_feature_map)
172  anchors.append(anchors_in_image)
173  anchors = [torch.cat(anchors_per_image) for anchors_per_image in anchors]
174  # Clear the cache in case that memory leaks.
175  self._cache.clear()
176  return anchors

179 class RPNHead(nn.Module):
180  """
181  Adds a simple RPN Head with classification and regression heads
182 
183  Arguments:
184  in_channels (int): number of channels of the input feature
185  num_anchors (int): number of anchors to be predicted
186  """
187 
188  def __init__(self, in_channels, num_anchors):
189  super(RPNHead, self).__init__()
190  self.conv = nn.Conv2d(
191  in_channels, in_channels, kernel_size=3, stride=1, padding=1
192  )
193  self.cls_logits = nn.Conv2d(in_channels, num_anchors, kernel_size=1, stride=1)
194  self.bbox_pred = nn.Conv2d(
195  in_channels, num_anchors * 4, kernel_size=1, stride=1
196  )
197 
198  for layer in self.children():
199  torch.nn.init.normal_(layer.weight, std=0.01)
200  torch.nn.init.constant_(layer.bias, 0)
201 
202  def forward(self, x):
203  # type: (List[Tensor]) -> Tuple[List[Tensor], List[Tensor]]
204  logits = []
205  bbox_reg = []
206  for feature in x:
207  t = F.relu(self.conv(feature))
208  logits.append(self.cls_logits(t))
209  bbox_reg.append(self.bbox_pred(t))
210  return logits, bbox_reg

213 def permute_and_flatten(layer, N, A, C, H, W):
214  # type: (Tensor, int, int, int, int, int) -> Tensor
215  layer = layer.view(N, -1, C, H, W)
216  layer = layer.permute(0, 3, 4, 1, 2)
217  layer = layer.reshape(N, -1, C)
218  return layer

221 def concat_box_prediction_layers(box_cls, box_regression):
222  # type: (List[Tensor], List[Tensor]) -> Tuple[Tensor, Tensor]
223  box_cls_flattened = []
224  box_regression_flattened = []
225  # for each feature level, permute the outputs to make them be in the
226  # same format as the labels. Note that the labels are computed for
227  # all feature levels concatenated, so we keep the same representation
228  # for the objectness and the box_regression
229  for box_cls_per_level, box_regression_per_level in zip(
230  box_cls, box_regression
231  ):
232  N, AxC, H, W = box_cls_per_level.shape
233  Ax4 = box_regression_per_level.shape[1]
234  A = Ax4 // 4
235  C = AxC // A
236  box_cls_per_level = permute_and_flatten(
237  box_cls_per_level, N, A, C, H, W
238  )
239  box_cls_flattened.append(box_cls_per_level)
240 
241  box_regression_per_level = permute_and_flatten(
242  box_regression_per_level, N, A, 4, H, W
243  )
244  box_regression_flattened.append(box_regression_per_level)
245  # concatenate on the first dimension (representing the feature levels), to
246  # take into account the way the labels were generated (with all feature maps
247  # being concatenated as well)
248  box_cls = torch.cat(box_cls_flattened, dim=1).flatten(0, -2)
249  box_regression = torch.cat(box_regression_flattened, dim=1).reshape(-1, 4)
250  return box_cls, box_regression

253 class RegionProposalNetwork(torch.nn.Module):
254  """
255  Implements Region Proposal Network (RPN).
256 
257  Arguments:
258  anchor_generator (AnchorGenerator): module that generates the anchors for a set of feature
259  maps.
260  head (nn.Module): module that computes the objectness and regression deltas
261  fg_iou_thresh (float): minimum IoU between the anchor and the GT box so that they can be
262  considered as positive during training of the RPN.
263  bg_iou_thresh (float): maximum IoU between the anchor and the GT box so that they can be
264  considered as negative during training of the RPN.
265  batch_size_per_image (int): number of anchors that are sampled during training of the RPN
266  for computing the loss
267  positive_fraction (float): proportion of positive anchors in a mini-batch during training
268  of the RPN
269  pre_nms_top_n (Dict[int]): number of proposals to keep before applying NMS. It should
270  contain two fields: training and testing, to allow for different values depending
271  on training or evaluation
272  post_nms_top_n (Dict[int]): number of proposals to keep after applying NMS. It should
273  contain two fields: training and testing, to allow for different values depending
274  on training or evaluation
275  nms_thresh (float): NMS threshold used for postprocessing the RPN proposals
276 
277  """
278  __annotations__ = {
279  'box_coder': det_utils.BoxCoder,
280  'proposal_matcher': det_utils.Matcher,
281  'fg_bg_sampler': det_utils.BalancedPositiveNegativeSampler,
282  'pre_nms_top_n': Dict[str, int],
283  'post_nms_top_n': Dict[str, int],
284  }
285 
286  def __init__(self,
287  anchor_generator,
288  head,
289  #
290  fg_iou_thresh, bg_iou_thresh,
291  batch_size_per_image, positive_fraction,
292  #
293  pre_nms_top_n, post_nms_top_n, nms_thresh):
294  super(RegionProposalNetwork, self).__init__()
295  self.anchor_generator = anchor_generator
296  self.head = head

327  def assign_targets_to_anchors(self, anchors, targets):
328  # type: (List[Tensor], List[Dict[str, Tensor]]) -> Tuple[List[Tensor], List[Tensor]]
329  labels = []
330  matched_gt_boxes = []
331  for anchors_per_image, targets_per_image in zip(anchors, targets):
332  gt_boxes = targets_per_image["boxes"]
333 
334  if gt_boxes.numel() == 0:
335  # Background image (negative example)
336  device = anchors_per_image.device
337  matched_gt_boxes_per_image = torch.zeros(anchors_per_image.shape, dtype=torch.float32, device=device)
338  labels_per_image = torch.zeros((anchors_per_image.shape[0],), dtype=torch.float32, device=device)
339  else:
340  match_quality_matrix = box_ops.box_iou(gt_boxes, anchors_per_image)
341  matched_idxs = self.proposal_matcher(match_quality_matrix)
342  # get the targets corresponding GT for each proposal
343  # NB: need to clamp the indices because we can have a single
344  # GT in the image, and matched_idxs can be -2, which goes
345  # out of bounds
346  matched_gt_boxes_per_image = gt_boxes[matched_idxs.clamp(min=0)]
347 
348  labels_per_image = matched_idxs >= 0
349  labels_per_image = labels_per_image.to(dtype=torch.float32)
350 
351  # Background (negative examples)
352  bg_indices = matched_idxs == self.proposal_matcher.BELOW_LOW_THRESHOLD
353  labels_per_image[bg_indices] = 0.0
354 
355  # discard indices that are between thresholds
356  inds_to_discard = matched_idxs == self.proposal_matcher.BETWEEN_THRESHOLDS
357  labels_per_image[inds_to_discard] = -1.0
358 
359  labels.append(labels_per_image)
360  matched_gt_boxes.append(matched_gt_boxes_per_image)
361  return labels, matched_gt_boxes

363  def _get_top_n_idx(self, objectness, num_anchors_per_level):
364  # type: (Tensor, List[int]) -> Tensor
365  r = []
366  offset = 0
367  for ob in objectness.split(num_anchors_per_level, 1):
368  if torchvision._is_tracing():
369  num_anchors, pre_nms_top_n = _onnx_get_num_anchors_and_pre_nms_top_n(ob, self.pre_nms_top_n())
370  else:
371  num_anchors = ob.shape[1]
372  pre_nms_top_n = min(self.pre_nms_top_n(), num_anchors)
373  _, top_n_idx = ob.topk(pre_nms_top_n, dim=1)
374  r.append(top_n_idx + offset)
375  offset += num_anchors
376  return torch.cat(r, dim=1)

378  def filter_proposals(self, proposals, objectness, image_shapes, num_anchors_per_level):
379  # type: (Tensor, Tensor, List[Tuple[int, int]], List[int]) -> Tuple[List[Tensor], List[Tensor]]
380  num_images = proposals.shape[0]
381  device = proposals.device
382  # do not backprop throught objectness
383  objectness = objectness.detach()
384  objectness = objectness.reshape(num_images, -1)
385 
386  levels = [
387  torch.full((n,), idx, dtype=torch.int64, device=device)
388  for idx, n in enumerate(num_anchors_per_level)
389  ]
390  levels = torch.cat(levels, 0)
391  levels = levels.reshape(1, -1).expand_as(objectness)
392 
393  # select top_n boxes independently per level before applying nms
394  top_n_idx = self._get_top_n_idx(objectness, num_anchors_per_level)
395 
396  image_range = torch.arange(num_images, device=device)
397  batch_idx = image_range[:, None]
398 
399  objectness = objectness[batch_idx, top_n_idx]
400  levels = levels[batch_idx, top_n_idx]
401  proposals = proposals[batch_idx, top_n_idx]
402 
403  final_boxes = []
404  final_scores = []
405  for boxes, scores, lvl, img_shape in zip(proposals, objectness, levels, image_shapes):
406  boxes = box_ops.clip_boxes_to_image(boxes, img_shape)
407  keep = box_ops.remove_small_boxes(boxes, self.min_size)
408  boxes, scores, lvl = boxes[keep], scores[keep], lvl[keep]
409  # non-maximum suppression, independently done per level
410  keep = box_ops.batched_nms(boxes, scores, lvl, self.nms_thresh)
411  # keep only topk scoring predictions
412  keep = keep[:self.post_nms_top_n()]
413  boxes, scores = boxes[keep], scores[keep]
414  final_boxes.append(boxes)
415  final_scores.append(scores)
416  return final_boxes, final_scores

418  def compute_loss(self, objectness, pred_bbox_deltas, labels, regression_targets):
419  # type: (Tensor, Tensor, List[Tensor], List[Tensor]) -> Tuple[Tensor, Tensor]
420  """
421  Arguments:
422  objectness (Tensor)
423  pred_bbox_deltas (Tensor)
424  labels (List[Tensor])
425  regression_targets (List[Tensor])
426 
427  Returns:
428  objectness_loss (Tensor)
429  box_loss (Tensor)
430  """
431 
432  sampled_pos_inds, sampled_neg_inds = self.fg_bg_sampler(labels)
433  sampled_pos_inds = torch.nonzero(torch.cat(sampled_pos_inds, dim=0)).squeeze(1)
434  sampled_neg_inds = torch.nonzero(torch.cat(sampled_neg_inds, dim=0)).squeeze(1)
435 
436  sampled_inds = torch.cat([sampled_pos_inds, sampled_neg_inds], dim=0)
437 
438  objectness = objectness.flatten()
439 
440  labels = torch.cat(labels, dim=0)
441  regression_targets = torch.cat(regression_targets, dim=0)
442 
443  box_loss = det_utils.smooth_l1_loss(
444  pred_bbox_deltas[sampled_pos_inds],
445  regression_targets[sampled_pos_inds],
446  beta=1 / 9,
447  size_average=False,
448  ) / (sampled_inds.numel())
449 
450  objectness_loss = F.binary_cross_entropy_with_logits(
451  objectness[sampled_inds], labels[sampled_inds]
452  )
453 
454  return objectness_loss, box_loss

456  def forward(self,
457  images, # type: ImageList
458  features, # type: Dict[str, Tensor]
459  targets=None # type: Optional[List[Dict[str, Tensor]]]
460  ):
461  # type: (...) -> Tuple[List[Tensor], Dict[str, Tensor]]
462  """
463  Arguments:
464  images (ImageList): images for which we want to compute the predictions
465  features (OrderedDict[Tensor]): features computed from the images that are
466  used for computing the predictions. Each tensor in the list
467  correspond to different feature levels
468  targets (List[Dict[Tensor]]): ground-truth boxes present in the image (optional).
469  If provided, each element in the dict should contain a field `boxes`,
470  with the locations of the ground-truth boxes.
471 
472  Returns:
473  boxes (List[Tensor]): the predicted boxes from the RPN, one Tensor per
474  image.
475  losses (Dict[Tensor]): the losses for the model during training. During
476  testing, it is an empty dict.
477  """
478  # RPN uses all feature maps that are available
479  features = list(features.values())
480  objectness, pred_bbox_deltas = self.head(features)
481  anchors = self.anchor_generator(images, features)
482 
483  num_images = len(anchors)
484  num_anchors_per_level_shape_tensors = [o[0].shape for o in objectness]
485  num_anchors_per_level = [s[0] * s[1] * s[2] for s in num_anchors_per_level_shape_tensors]
486  objectness, pred_bbox_deltas = \
487  concat_box_prediction_layers(objectness, pred_bbox_deltas)
488  # apply pred_bbox_deltas to anchors to obtain the decoded proposals
489  # note that we detach the deltas because Faster R-CNN do not backprop through
490  # the proposals
491  proposals = self.box_coder.decode(pred_bbox_deltas.detach(), anchors)
492  proposals = proposals.view(num_images, -1, 4)
493  boxes, scores = self.filter_proposals(proposals, objectness, images.image_sizes, num_anchors_per_level)
494 
495  losses = {}
496  if self.training:
497  assert targets is not None
498  labels, matched_gt_boxes = self.assign_targets_to_anchors(anchors, targets)
499  regression_targets = self.box_coder.encode(matched_gt_boxes, anchors)
500  loss_objectness, loss_rpn_box_reg = self.compute_loss(
501  objectness, pred_bbox_deltas, labels, regression_targets)
502  losses = {
503  "loss_objectness": loss_objectness,
504  "loss_rpn_box_reg": loss_rpn_box_reg,
505  }
506  return boxes, losses
