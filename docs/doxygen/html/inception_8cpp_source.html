302 InceptionV3Output InceptionV3Impl::forward(torch::Tensor x) {
303  if (transform_input) {
304  auto x_ch0 = torch::unsqueeze(x.select(1, 0), 1) * (0.229 / 0.5) +
305  (0.485 - 0.5) / 0.5;
306  auto x_ch1 = torch::unsqueeze(x.select(1, 1), 1) * (0.224 / 0.5) +
307  (0.456 - 0.5) / 0.5;
308  auto x_ch2 = torch::unsqueeze(x.select(1, 2), 1) * (0.225 / 0.5) +
309  (0.406 - 0.5) / 0.5;
310 
311  x = torch::cat({x_ch0, x_ch1, x_ch2}, 1);
312  }
313 
314  // N x 3 x 299 x 299
315  x = Conv2d_1a_3x3->forward(x);
316  // N x 32 x 149 x 149
317  x = Conv2d_2a_3x3->forward(x);
318  // N x 32 x 147 x 147
319  x = Conv2d_2b_3x3->forward(x);
320  // N x 64 x 147 x 147
321  x = torch::max_pool2d(x, 3, 2);
322  // N x 64 x 73 x 73
323  x = Conv2d_3b_1x1->forward(x);
324  // N x 80 x 73 x 73
325  x = Conv2d_4a_3x3->forward(x);
326  // N x 192 x 71 x 71
327  x = torch::max_pool2d(x, 3, 2);
328  // N x 192 x 35 x 35
329  x = Mixed_5b->forward(x);
330  // N x 256 x 35 x 35
331  x = Mixed_5c->forward(x);
332  // N x 288 x 35 x 35
333  x = Mixed_5d->forward(x);
334  // N x 288 x 35 x 35
335  x = Mixed_6a->forward(x);
336  // N x 768 x 17 x 17
337  x = Mixed_6b->forward(x);
338  // N x 768 x 17 x 17
339  x = Mixed_6c->forward(x);
340  // N x 768 x 17 x 17
341  x = Mixed_6d->forward(x);
342  // N x 768 x 17 x 17
343  x = Mixed_6e->forward(x);
344  // N x 768 x 17 x 17
345 
346  torch::Tensor aux;
347  if (is_training() && aux_logits)
348  aux = AuxLogits->forward(x);
349 
350  // N x 768 x 17 x 17
351  x = Mixed_7a->forward(x);
352  // N x 1280 x 8 x 8
353  x = Mixed_7b->forward(x);
354  // N x 2048 x 8 x 8
355  x = Mixed_7c->forward(x);

357  x = torch::adaptive_avg_pool2d(x, {1, 1});
358  // N x 2048 x 1 x 1

360  // N x 2048 x 1 x 1
361  x = x.view({x.size(0), -1});
362  // N x 2048
363  x = fc->forward(x);
364  // N x 1000 (num_classes)
365 
366  if (is_training() && aux_logits)
367  return {x, aux};
368  return {x, {}};
