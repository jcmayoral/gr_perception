28  T variance(const std::list<T> &li, T mean)
29  {
30  size_t sz = li.size();
31  if (sz == 1)
32  return 0.0;
33  // Calculate the mean
34  //U mean_calc = std::accumulate(li.begin(), li.end(), 0.0) / sz;
35  // Now calculate the variance
36  auto variance_func = [&mean, &sz](T accumulator, const T& val)
37  {
38  return accumulator + ((val - mean)*(val - mean) / (sz - 1));
39  };
40  return std::accumulate(li.begin(), li.end(), 0.0, variance_func);
41  }
