27 def inception_v3(pretrained=False, progress=True, **kwargs):
28  r"""Inception v3 model architecture from
29  `"Rethinking the Inception Architecture for Computer Vision" <http://arxiv.org/abs/1512.00567>`_.
30 
31  .. note::
32  **Important**: In contrast to the other models the inception_v3 expects tensors with a size of
33  N x 3 x 299 x 299, so ensure your images are sized accordingly.
34 
35  Args:
36  pretrained (bool): If True, returns a model pre-trained on ImageNet
37  progress (bool): If True, displays a progress bar of the download to stderr
38  aux_logits (bool): If True, add an auxiliary branch that can improve training.
39  Default: *True*
40  transform_input (bool): If True, preprocesses the input according to the method with which it
41  was trained on ImageNet. Default: *False*
42  """
43  if pretrained:
44  if 'transform_input' not in kwargs:
45  kwargs['transform_input'] = True
46  if 'aux_logits' in kwargs:
47  original_aux_logits = kwargs['aux_logits']
48  kwargs['aux_logits'] = True
49  else:
50  original_aux_logits = True
51  kwargs['init_weights'] = False # we are loading weights from a pretrained model
52  model = Inception3(**kwargs)
53  state_dict = load_state_dict_from_url(model_urls['inception_v3_google'],
54  progress=progress)
55  model.load_state_dict(state_dict)
56  if not original_aux_logits:
57  model.aux_logits = False
58  del model.AuxLogits
59  return model
60 
61  return Inception3(**kwargs)

102  self.Mixed_6c = inception_c(768, channels_7x7=160)
103  self.Mixed_6d = inception_c(768, channels_7x7=160)
104  self.Mixed_6e = inception_c(768, channels_7x7=192)
105  if aux_logits:

107  self.Mixed_7a = inception_d(768)
108  self.Mixed_7b = inception_e(1280)
109  self.Mixed_7c = inception_e(2048)

111  self.dropout = nn.Dropout()

113  if init_weights:
114  for m in self.modules():
115  if isinstance(m, nn.Conv2d) or isinstance(m, nn.Linear):
116  import scipy.stats as stats
117  stddev = m.stddev if hasattr(m, 'stddev') else 0.1
118  X = stats.truncnorm(-2, 2, scale=stddev)
119  values = torch.as_tensor(X.rvs(m.weight.numel()), dtype=m.weight.dtype)
120  values = values.view(m.weight.size())
121  with torch.no_grad():
122  m.weight.copy_(values)
123  elif isinstance(m, nn.BatchNorm2d):

125  nn.init.constant_(m.bias, 0)

127  def _transform_input(self, x):
128  if self.transform_input:
129  x_ch0 = torch.unsqueeze(x[:, 0], 1) * (0.229 / 0.5) + (0.485 - 0.5) / 0.5
130  x_ch1 = torch.unsqueeze(x[:, 1], 1) * (0.224 / 0.5) + (0.456 - 0.5) / 0.5
131  x_ch2 = torch.unsqueeze(x[:, 2], 1) * (0.225 / 0.5) + (0.406 - 0.5) / 0.5
132  x = torch.cat((x_ch0, x_ch1, x_ch2), 1)

155  x = self.Mixed_5d(x)

157  x = self.Mixed_6a(x)
158  # N x 768 x 17 x 17

161  x = self.Mixed_6c(x)

164  # N x 768 x 17 x 17

171  aux = None
172  # N x 768 x 17 x 17
173  x = self.Mixed_7a(x)

177  x = self.Mixed_7c(x)

183  # N x 2048 x 1 x 1
