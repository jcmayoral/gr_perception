26 def googlenet(pretrained=False, progress=True, **kwargs):
27  r"""GoogLeNet (Inception v1) model architecture from
28  `"Going Deeper with Convolutions" <http://arxiv.org/abs/1409.4842>`_.
29 
30  Args:
31  pretrained (bool): If True, returns a model pre-trained on ImageNet
32  progress (bool): If True, displays a progress bar of the download to stderr
33  aux_logits (bool): If True, adds two auxiliary branches that can improve training.
34  Default: *False* when pretrained is True otherwise *True*
35  transform_input (bool): If True, preprocesses the input according to the method with which it
36  was trained on ImageNet. Default: *False*
37  """
38  if pretrained:
39  if 'transform_input' not in kwargs:
40  kwargs['transform_input'] = True
41  if 'aux_logits' not in kwargs:
42  kwargs['aux_logits'] = False
43  if kwargs['aux_logits']:
44  warnings.warn('auxiliary heads in the pretrained googlenet model are NOT pretrained, '
45  'so make sure to train them')
46  original_aux_logits = kwargs['aux_logits']
47  kwargs['aux_logits'] = True
48  kwargs['init_weights'] = False
49  model = GoogLeNet(**kwargs)
50  state_dict = load_state_dict_from_url(model_urls['googlenet'],
51  progress=progress)
52  model.load_state_dict(state_dict)
53  if not original_aux_logits:
54  model.aux_logits = False
55  model.aux1 = None
56  model.aux2 = None
57  return model

59  return GoogLeNet(**kwargs)

218  def __init__(self, in_channels, ch1x1, ch3x3red, ch3x3, ch5x5red, ch5x5, pool_proj,
219  conv_block=None):
220  super(Inception, self).__init__()
221  if conv_block is None:
222  conv_block = BasicConv2d
223  self.branch1 = conv_block(in_channels, ch1x1, kernel_size=1)
224 
225  self.branch2 = nn.Sequential(
226  conv_block(in_channels, ch3x3red, kernel_size=1),
227  conv_block(ch3x3red, ch3x3, kernel_size=3, padding=1)
228  )
229 
230  self.branch3 = nn.Sequential(
231  conv_block(in_channels, ch5x5red, kernel_size=1),
232  # Here, kernel_size=3 instead of kernel_size=5 is a known bug.
233  # Please see https://github.com/pytorch/vision/issues/906 for details.
234  conv_block(ch5x5red, ch5x5, kernel_size=3, padding=1)
235  )
236 
237  self.branch4 = nn.Sequential(
238  nn.MaxPool2d(kernel_size=3, stride=1, padding=1, ceil_mode=True),
239  conv_block(in_channels, pool_proj, kernel_size=1)
240  )
