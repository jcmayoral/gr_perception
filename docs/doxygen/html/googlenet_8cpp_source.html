166  if (transform_input) {
167  auto x_ch0 = torch::unsqueeze(x.select(1, 0), 1) * (0.229 / 0.5) +
168  (0.485 - 0.5) / 0.5;
169  auto x_ch1 = torch::unsqueeze(x.select(1, 1), 1) * (0.224 / 0.5) +
170  (0.456 - 0.5) / 0.5;
171  auto x_ch2 = torch::unsqueeze(x.select(1, 2), 1) * (0.225 / 0.5) +
172  (0.406 - 0.5) / 0.5;
173 
174  x = torch::cat({x_ch0, x_ch1, x_ch2}, 1);
175  }
176 
177  // N x 3 x 224 x 224
178  x = conv1->forward(x);
179  // N x 64 x 112 x 112
180  x = torch::max_pool2d(x, 3, 2, 0, 1, true);
181  // N x 64 x 56 x 56
182  x = conv2->forward(x);
183  // N x 64 x 56 x 56
184  x = conv3->forward(x);
185  // N x 192 x 56 x 56
186  x = torch::max_pool2d(x, 3, 2, 0, 1, true);
187 
188  // N x 192 x 28 x 28
189  x = inception3a->forward(x);
190  // N x 256 x 28 x 28
191  x = inception3b->forward(x);
192  // N x 480 x 28 x 28
193  x = torch::max_pool2d(x, 3, 2, 0, 1, true);
194  // N x 480 x 14 x 14
195  x = inception4a->forward(x);
196  // N x 512 x 14 x 14
197  torch::Tensor aux1;
198  if (is_training() && aux_logits)
199  aux1 = this->aux1->forward(x);
200 
201  x = inception4b->forward(x);
202  // N x 512 x 14 x 14
203  x = inception4c->forward(x);
204  // N x 512 x 14 x 14

206  // N x 528 x 14 x 14
207  torch::Tensor aux2;
