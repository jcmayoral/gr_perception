106  def conv2d(layer_input, filters, f_size=4, bn=True):
107  """Layers used during downsampling"""
108  d = Conv2D(filters, kernel_size=f_size, strides=2, padding='same')(layer_input)

114  def deconv2d(layer_input, skip_input, filters, f_size=4, dropout_rate=0):
115  """Layers used during upsampling"""
116  u = UpSampling2D(size=2)(layer_input)
117  u = Conv2D(filters, kernel_size=f_size, strides=1, padding='same', activation='relu')(u)
118  if dropout_rate:
119  u = Dropout(dropout_rate)(u)
120  u = BatchNormalization(momentum=0.8)(u)
121  u = Concatenate()([u, skip_input])
122  return u
