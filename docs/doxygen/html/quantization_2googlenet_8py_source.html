22 def googlenet(pretrained=False, progress=True, quantize=False, **kwargs):
23  r"""GoogLeNet (Inception v1) model architecture from
24  `"Going Deeper with Convolutions" <http://arxiv.org/abs/1409.4842>`_.
25 
26  Note that quantize = True returns a quantized model with 8 bit
27  weights. Quantized models only support inference and run on CPUs.
28  GPU inference is not yet supported
29 
30  Args:
31  pretrained (bool): If True, returns a model pre-trained on ImageNet
32  progress (bool): If True, displays a progress bar of the download to stderr
33  aux_logits (bool): If True, adds two auxiliary branches that can improve training.
34  Default: *False* when pretrained is True otherwise *True*
35  transform_input (bool): If True, preprocesses the input according to the method with which it
36  was trained on ImageNet. Default: *False*
37  """
38  if pretrained:
39  if 'transform_input' not in kwargs:
40  kwargs['transform_input'] = True
41  if 'aux_logits' not in kwargs:
42  kwargs['aux_logits'] = False
43  if kwargs['aux_logits']:
44  warnings.warn('auxiliary heads in the pretrained googlenet model are NOT pretrained, '
45  'so make sure to train them')
46  original_aux_logits = kwargs['aux_logits']
47  kwargs['aux_logits'] = True
48  kwargs['init_weights'] = False
49 
50  model = QuantizableGoogLeNet(**kwargs)
51  _replace_relu(model)
52 
53  if quantize:
54  # TODO use pretrained as a string to specify the backend
55  backend = 'fbgemm'
56  quantize_model(model, backend)
57  else:
58  assert pretrained in [True, False]
59 
60  if pretrained:
61  if quantize:
62  model_url = quant_model_urls['googlenet' + '_' + backend]
63  else:
64  model_url = model_urls['googlenet']
65 
66  state_dict = load_state_dict_from_url(model_url,
67  progress=progress)
68 
69  model.load_state_dict(state_dict)
70 
71  if not original_aux_logits:
72  model.aux_logits = False
73  model.aux1 = None
74  model.aux2 = None
75  return model
