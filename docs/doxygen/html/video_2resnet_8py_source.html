193  def __init__(self, block, conv_makers, layers,
194  stem, num_classes=400,
195  zero_init_residual=False):
196  """Generic resnet video generator.
197 
198  Args:
199  block (nn.Module): resnet building block
200  conv_makers (list(functions)): generator function for each layer
201  layers (List[int]): number of blocks per layer
202  stem (nn.Module, optional): Resnet stem, if None, defaults to conv-bn-relu. Defaults to None.
203  num_classes (int, optional): Dimension of the final FC layer. Defaults to 400.
204  zero_init_residual (bool, optional): Zero init bottleneck residual BN. Defaults to False.
205  """
206  super(VideoResNet, self).__init__()
207  self.inplanes = 64
208 
209  self.stem = stem()
210 
211  self.layer1 = self._make_layer(block, conv_makers[0], 64, layers[0], stride=1)
212  self.layer2 = self._make_layer(block, conv_makers[1], 128, layers[1], stride=2)
213  self.layer3 = self._make_layer(block, conv_makers[2], 256, layers[2], stride=2)
214  self.layer4 = self._make_layer(block, conv_makers[3], 512, layers[3], stride=2)
215 
216  self.avgpool = nn.AdaptiveAvgPool3d((1, 1, 1))
217  self.fc = nn.Linear(512 * block.expansion, num_classes)
218 
219  # init weights
220  self._initialize_weights()
221 
222  if zero_init_residual:
223  for m in self.modules():
224  if isinstance(m, Bottleneck):
225  nn.init.constant_(m.bn3.weight, 0)
226 
227  def forward(self, x):
228  x = self.stem(x)
229 
230  x = self.layer1(x)
231  x = self.layer2(x)
232  x = self.layer3(x)
233  x = self.layer4(x)
234 
235  x = self.avgpool(x)
236  # Flatten the layer to fc
237  x = x.flatten(1)
238  x = self.fc(x)
239 
240  return x

242  def _make_layer(self, block, conv_builder, planes, blocks, stride=1):
243  downsample = None
244 
245  if stride != 1 or self.inplanes != planes * block.expansion:
246  ds_stride = conv_builder.get_downsample_stride(stride)
247  downsample = nn.Sequential(
248  nn.Conv3d(self.inplanes, planes * block.expansion,
249  kernel_size=1, stride=ds_stride, bias=False),
250  nn.BatchNorm3d(planes * block.expansion)
251  )
252  layers = []
253  layers.append(block(self.inplanes, planes, conv_builder, stride, downsample))
254 
255  self.inplanes = planes * block.expansion
256  for i in range(1, blocks):
257  layers.append(block(self.inplanes, planes, conv_builder))
258 
259  return nn.Sequential(*layers)

261  def _initialize_weights(self):
262  for m in self.modules():
263  if isinstance(m, nn.Conv3d):
264  nn.init.kaiming_normal_(m.weight, mode='fan_out',
265  nonlinearity='relu')

269  nn.init.constant_(m.weight, 1)
270  nn.init.constant_(m.bias, 0)
