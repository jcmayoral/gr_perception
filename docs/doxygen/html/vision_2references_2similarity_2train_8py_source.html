41 def find_best_threshold(dists, targets, device):
42  best_thresh = 0.01
43  best_correct = 0
44  for thresh in torch.arange(0.0, 1.51, 0.01):
45  predictions = dists <= thresh.to(device)
46  correct = torch.sum(predictions == targets.to(device)).item()
47  if correct > best_correct:
48  best_thresh = thresh
49  best_correct = correct
50 
51  accuracy = 100.0 * best_correct / dists.size(0)
52 
53  return best_thresh, accuracy
