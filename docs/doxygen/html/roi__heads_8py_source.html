16 def fastrcnn_loss(class_logits, box_regression, labels, regression_targets):
17  # type: (Tensor, Tensor, List[Tensor], List[Tensor]) -> Tuple[Tensor, Tensor]
18  """
19  Computes the loss for Faster R-CNN.
20 
21  Arguments:
22  class_logits (Tensor)
23  box_regression (Tensor)
24  labels (list[BoxList])
25  regression_targets (Tensor)
26 
27  Returns:
28  classification_loss (Tensor)
29  box_loss (Tensor)
30  """
31 
32  labels = torch.cat(labels, dim=0)
33  regression_targets = torch.cat(regression_targets, dim=0)
34 
35  classification_loss = F.cross_entropy(class_logits, labels)
36 
37  # get indices that correspond to the regression targets for
38  # the corresponding ground truth labels, to be used with
39  # advanced indexing
40  sampled_pos_inds_subset = torch.nonzero(labels > 0).squeeze(1)
41  labels_pos = labels[sampled_pos_inds_subset]
42  N, num_classes = class_logits.shape
43  box_regression = box_regression.reshape(N, -1, 4)
44

46  box_regression[sampled_pos_inds_subset, labels_pos],
47  regression_targets[sampled_pos_inds_subset],
48  beta=1 / 9,
49  size_average=False,
50  )
51  box_loss = box_loss / labels.numel()
52 
53  return classification_loss, box_loss

56 def maskrcnn_inference(x, labels):
57  # type: (Tensor, List[Tensor]) -> List[Tensor]
58  """
59  From the results of the CNN, post process the masks
60  by taking the mask corresponding to the class with max
61  probability (which are of fixed size and directly output
62  by the CNN) and return the masks in the mask field of the BoxList.
63 
64  Arguments:
65  x (Tensor): the mask logits
66  labels (list[BoxList]): bounding boxes that are used as
67  reference, one for ech image
68 
69  Returns:
70  results (list[BoxList]): one BoxList for each image, containing
71  the extra field mask
72  """
73  mask_prob = x.sigmoid()
74 
75  # select masks corresponding to the predicted classes
76  num_masks = x.shape[0]
77  boxes_per_image = [label.shape[0] for label in labels]
78  labels = torch.cat(labels)
79  index = torch.arange(num_masks, device=labels.device)
80  mask_prob = mask_prob[index, labels][:, None]
81  mask_prob = mask_prob.split(boxes_per_image, dim=0)
82 
83  return mask_prob

134 def keypoints_to_heatmap(keypoints, rois, heatmap_size):
135  # type: (Tensor, Tensor, int) -> Tuple[Tensor, Tensor]
136  offset_x = rois[:, 0]
137  offset_y = rois[:, 1]
138  scale_x = heatmap_size / (rois[:, 2] - rois[:, 0])
139  scale_y = heatmap_size / (rois[:, 3] - rois[:, 1])
140 
141  offset_x = offset_x[:, None]
142  offset_y = offset_y[:, None]
143  scale_x = scale_x[:, None]
144  scale_y = scale_y[:, None]
145 
146  x = keypoints[..., 0]
147  y = keypoints[..., 1]
148 
149  x_boundary_inds = x == rois[:, 2][:, None]
150  y_boundary_inds = y == rois[:, 3][:, None]
151 
152  x = (x - offset_x) * scale_x
153  x = x.floor().long()
154  y = (y - offset_y) * scale_y
155  y = y.floor().long()
156 
157  x[x_boundary_inds] = heatmap_size - 1
158  y[y_boundary_inds] = heatmap_size - 1
159 
160  valid_loc = (x >= 0) & (y >= 0) & (x < heatmap_size) & (y < heatmap_size)
161  vis = keypoints[..., 2] > 0
162  valid = (valid_loc & vis).long()
163 
164  lin_ind = y * heatmap_size + x
165  heatmaps = lin_ind * valid
166 
167  return heatmaps, valid

170 def _onnx_heatmaps_to_keypoints(maps, maps_i, roi_map_width, roi_map_height,
171  widths_i, heights_i, offset_x_i, offset_y_i):
172  num_keypoints = torch.scalar_tensor(maps.size(1), dtype=torch.int64)
173 
174  width_correction = widths_i / roi_map_width
175  height_correction = heights_i / roi_map_height
176 
177  roi_map = F.interpolate(
178  maps_i[:, None], size=(int(roi_map_height), int(roi_map_width)), mode='bicubic', align_corners=False)[:, 0]
179 
180  w = torch.scalar_tensor(roi_map.size(2), dtype=torch.int64)
181  pos = roi_map.reshape(num_keypoints, -1).argmax(dim=1)
182 
183  x_int = (pos % w)
184  y_int = ((pos - x_int) // w)
185 
186  x = (torch.tensor(0.5, dtype=torch.float32) + x_int.to(dtype=torch.float32)) * \
187  width_correction.to(dtype=torch.float32)
188  y = (torch.tensor(0.5, dtype=torch.float32) + y_int.to(dtype=torch.float32)) * \
189  height_correction.to(dtype=torch.float32)
190 
191  xy_preds_i_0 = x + offset_x_i.to(dtype=torch.float32)
192  xy_preds_i_1 = y + offset_y_i.to(dtype=torch.float32)
193  xy_preds_i_2 = torch.ones((xy_preds_i_1.shape), dtype=torch.float32)
194  xy_preds_i = torch.stack([xy_preds_i_0.to(dtype=torch.float32),
195  xy_preds_i_1.to(dtype=torch.float32),
196  xy_preds_i_2.to(dtype=torch.float32)], 0)
197 
198  # TODO: simplify when indexing without rank will be supported by ONNX
199  base = num_keypoints * num_keypoints + num_keypoints + 1
200  ind = torch.arange(num_keypoints)
201  ind = ind.to(dtype=torch.int64) * base
202  end_scores_i = roi_map.index_select(1, y_int.to(dtype=torch.int64)) \
203  .index_select(2, x_int.to(dtype=torch.int64)).view(-1).index_select(0, ind.to(dtype=torch.int64))
204 
205  return xy_preds_i, end_scores_i

226 def heatmaps_to_keypoints(maps, rois):
227  """Extract predicted keypoint locations from heatmaps. Output has shape
228  (#rois, 4, #keypoints) with the 4 rows corresponding to (x, y, logit, prob)
229  for each keypoint.
230  """
231  # This function converts a discrete image coordinate in a HEATMAP_SIZE x
232  # HEATMAP_SIZE image to a continuous keypoint coordinate. We maintain
233  # consistency with keypoints_to_heatmap_labels by using the conversion from
234  # Heckbert 1990: c = d + 0.5, where d is a discrete coordinate and c is a
235  # continuous coordinate.
236  offset_x = rois[:, 0]
237  offset_y = rois[:, 1]
238 
239  widths = rois[:, 2] - rois[:, 0]
240  heights = rois[:, 3] - rois[:, 1]
241  widths = widths.clamp(min=1)
242  heights = heights.clamp(min=1)
243  widths_ceil = widths.ceil()
244  heights_ceil = heights.ceil()
245 
246  num_keypoints = maps.shape[1]
247 
248  if torchvision._is_tracing():
249  xy_preds, end_scores = _onnx_heatmaps_to_keypoints_loop(maps, rois,
250  widths_ceil, heights_ceil, widths, heights,
251  offset_x, offset_y,
252  torch.scalar_tensor(num_keypoints, dtype=torch.int64))
253  return xy_preds.permute(0, 2, 1), end_scores
254 
255  xy_preds = torch.zeros((len(rois), 3, num_keypoints), dtype=torch.float32, device=maps.device)
256  end_scores = torch.zeros((len(rois), num_keypoints), dtype=torch.float32, device=maps.device)
257  for i in range(len(rois)):
258  roi_map_width = int(widths_ceil[i].item())
259  roi_map_height = int(heights_ceil[i].item())
260  width_correction = widths[i] / roi_map_width
261  height_correction = heights[i] / roi_map_height
262  roi_map = F.interpolate(
263  maps[i][:, None], size=(roi_map_height, roi_map_width), mode='bicubic', align_corners=False)[:, 0]
264  # roi_map_probs = scores_to_probs(roi_map.copy())
265  w = roi_map.shape[2]
266  pos = roi_map.reshape(num_keypoints, -1).argmax(dim=1)
267 
268  x_int = pos % w
269  y_int = (pos - x_int) // w
270  # assert (roi_map_probs[k, y_int, x_int] ==
271  # roi_map_probs[k, :, :].max())
272  x = (x_int.float() + 0.5) * width_correction
273  y = (y_int.float() + 0.5) * height_correction
274  xy_preds[i, 0, :] = x + offset_x[i]
275  xy_preds[i, 1, :] = y + offset_y[i]
276  xy_preds[i, 2, :] = 1
277  end_scores[i, :] = roi_map[torch.arange(num_keypoints), y_int, x_int]
278 
279  return xy_preds.permute(0, 2, 1), end_scores

282 def keypointrcnn_loss(keypoint_logits, proposals, gt_keypoints, keypoint_matched_idxs):
283  # type: (Tensor, List[Tensor], List[Tensor], List[Tensor]) -> Tensor
284  N, K, H, W = keypoint_logits.shape
285  assert H == W
286  discretization_size = H
287  heatmaps = []
288  valid = []
289  for proposals_per_image, gt_kp_in_image, midx in zip(proposals, gt_keypoints, keypoint_matched_idxs):
290  kp = gt_kp_in_image[midx]
291  heatmaps_per_image, valid_per_image = keypoints_to_heatmap(
292  kp, proposals_per_image, discretization_size
293  )
294  heatmaps.append(heatmaps_per_image.view(-1))
295  valid.append(valid_per_image.view(-1))
296 
297  keypoint_targets = torch.cat(heatmaps, dim=0)
298  valid = torch.cat(valid, dim=0).to(dtype=torch.uint8)
299  valid = torch.nonzero(valid).squeeze(1)
300 
301  # torch.mean (in binary_cross_entropy_with_logits) does'nt
302  # accept empty tensors, so handle it sepaartely
303  if keypoint_targets.numel() == 0 or len(valid) == 0:
304  return keypoint_logits.sum() * 0
305 
306  keypoint_logits = keypoint_logits.view(N * K, H * W)
307 
308  keypoint_loss = F.cross_entropy(keypoint_logits[valid], keypoint_targets[valid])
309  return keypoint_loss

328 def _onnx_expand_boxes(boxes, scale):
329  # type: (Tensor, float) -> Tensor
330  w_half = (boxes[:, 2] - boxes[:, 0]) * .5
331  h_half = (boxes[:, 3] - boxes[:, 1]) * .5
332  x_c = (boxes[:, 2] + boxes[:, 0]) * .5
333  y_c = (boxes[:, 3] + boxes[:, 1]) * .5
334

336  h_half = h_half.to(dtype=torch.float32) * scale
337 
338  boxes_exp0 = x_c - w_half
339  boxes_exp1 = y_c - h_half
340  boxes_exp2 = x_c + w_half
341  boxes_exp3 = y_c + h_half
342  boxes_exp = torch.stack((boxes_exp0, boxes_exp1, boxes_exp2, boxes_exp3), 1)
343  return boxes_exp

386 def paste_mask_in_image(mask, box, im_h, im_w):
387  # type: (Tensor, Tensor, int, int) -> Tensor
388  TO_REMOVE = 1
389  w = int(box[2] - box[0] + TO_REMOVE)
390  h = int(box[3] - box[1] + TO_REMOVE)
391  w = max(w, 1)
392  h = max(h, 1)

394  # Set shape to [batchxCxHxW]
395  mask = mask.expand((1, 1, -1, -1))

397  # Resize mask

399  mask = mask[0][0]

402  x_0 = max(box[0], 0)
403  x_1 = min(box[2] + 1, im_w)

405  y_1 = min(box[3] + 1, im_h)

413 def _onnx_paste_mask_in_image(mask, box, im_h, im_w):
414  one = torch.ones(1, dtype=torch.int64)
415  zero = torch.zeros(1, dtype=torch.int64)
416 
417  w = (box[2] - box[0] + one)
418  h = (box[3] - box[1] + one)
419  w = torch.max(torch.cat((w, one)))
420  h = torch.max(torch.cat((h, one)))
421 
422  # Set shape to [batchxCxHxW]
423  mask = mask.expand((1, 1, mask.size(0), mask.size(1)))
424 
425  # Resize mask
426  mask = F.interpolate(mask, size=(int(h), int(w)), mode='bilinear', align_corners=False)
427  mask = mask[0][0]
428 
429  x_0 = torch.max(torch.cat((box[0].unsqueeze(0), zero)))
430  x_1 = torch.min(torch.cat((box[2].unsqueeze(0) + one, im_w.unsqueeze(0))))
431  y_0 = torch.max(torch.cat((box[1].unsqueeze(0), zero)))
432  y_1 = torch.min(torch.cat((box[3].unsqueeze(0) + one, im_h.unsqueeze(0))))
433 
434  unpaded_im_mask = mask[(y_0 - box[1]):(y_1 - box[1]),
435  (x_0 - box[0]):(x_1 - box[0])]
436 
437  # TODO : replace below with a dynamic padding when support is added in ONNX
438 
439  # pad y
440  zeros_y0 = torch.zeros(y_0, unpaded_im_mask.size(1))
441  zeros_y1 = torch.zeros(im_h - y_1, unpaded_im_mask.size(1))
442  concat_0 = torch.cat((zeros_y0,
443  unpaded_im_mask.to(dtype=torch.float32),
444  zeros_y1), 0)[0:im_h, :]
445  # pad x
446  zeros_x0 = torch.zeros(concat_0.size(0), x_0)
447  zeros_x1 = torch.zeros(concat_0.size(0), im_w - x_1)
448  im_mask = torch.cat((zeros_x0,
449  concat_0,
450  zeros_x1), 1)[:, :im_w]
451  return im_mask
