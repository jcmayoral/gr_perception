119 @torch.jit.unused
120 def adjust_hue(img, hue_factor):
121  """Adjust hue of an image.
122 
123  The image hue is adjusted by converting the image to HSV and
124  cyclically shifting the intensities in the hue channel (H).
125  The image is then converted back to original image mode.
126 
127  `hue_factor` is the amount of shift in H channel and must be in the
128  interval `[-0.5, 0.5]`.
129 
130  See `Hue`_ for more details.
131 
132  .. _Hue: https://en.wikipedia.org/wiki/Hue
133 
134  Args:
135  img (PIL Image): PIL Image to be adjusted.
136  hue_factor (float): How much to shift the hue channel. Should be in
137  [-0.5, 0.5]. 0.5 and -0.5 give complete reversal of hue channel in
138  HSV space in positive and negative direction respectively.
139  0 means no shift. Therefore, both -0.5 and 0.5 will give an image
140  with complementary colors while 0 gives the original image.
141 
142  Returns:
143  PIL Image: Hue adjusted image.
144  """
145  if not(-0.5 <= hue_factor <= 0.5):
146  raise ValueError('hue_factor ({}) is not in [-0.5, 0.5].'.format(hue_factor))
147 
148  if not _is_pil_image(img):
149  raise TypeError('img should be PIL Image. Got {}'.format(type(img)))
150 
151  input_mode = img.mode
152  if input_mode in {'L', '1', 'I', 'F'}:
153  return img
154 
155  h, s, v = img.convert('HSV').split()
156 
157  np_h = np.array(h, dtype=np.uint8)
158  # uint8 addition take cares of rotation across boundaries
159  with np.errstate(over='ignore'):
160  np_h += np.uint8(hue_factor * 255)
161  h = Image.fromarray(np_h, 'L')
162 
163  img = Image.merge('HSV', (h, s, v)).convert(input_mode)
164  return img

167 @torch.jit.unused
168 def pad(img, padding, fill=0, padding_mode="constant"):
169  r"""Pad the given PIL.Image on all sides with the given "pad" value.
170 
171  Args:
172  img (PIL Image): Image to be padded.
173  padding (int or tuple or list): Padding on each border. If a single int is provided this
174  is used to pad all borders. If a tuple or list of length 2 is provided this is the padding
175  on left/right and top/bottom respectively. If a tuple or list of length 4 is provided
176  this is the padding for the left, top, right and bottom borders respectively. For compatibility reasons
177  with ``functional_tensor.pad``, if a tuple or list of length 1 is provided, it is interpreted as
178  a single int.
179  fill (int or str or tuple): Pixel fill value for constant fill. Default is 0. If a tuple of
180  length 3, it is used to fill R, G, B channels respectively.
181  This value is only used when the padding_mode is constant.
182  padding_mode: Type of padding. Should be: constant, edge, reflect or symmetric. Default is constant.
183 
184  - constant: pads with a constant value, this value is specified with fill
185 
186  - edge: pads with the last value on the edge of the image
187 
188  - reflect: pads with reflection of image (without repeating the last value on the edge)
189 
190  padding [1, 2, 3, 4] with 2 elements on both sides in reflect mode
191  will result in [3, 2, 1, 2, 3, 4, 3, 2]
192 
193  - symmetric: pads with reflection of image (repeating the last value on the edge)
194 
195  padding [1, 2, 3, 4] with 2 elements on both sides in symmetric mode
196  will result in [2, 1, 1, 2, 3, 4, 4, 3]
197 
198  Returns:
199  PIL Image: Padded image.
200  """
201 
202  if not _is_pil_image(img):
203  raise TypeError("img should be PIL Image. Got {}".format(type(img)))
204 
205  if not isinstance(padding, (numbers.Number, tuple, list)):
206  raise TypeError("Got inappropriate padding arg")
207  if not isinstance(fill, (numbers.Number, str, tuple)):
208  raise TypeError("Got inappropriate fill arg")
209  if not isinstance(padding_mode, str):
210  raise TypeError("Got inappropriate padding_mode arg")
211 
212  if isinstance(padding, list):
213  padding = tuple(padding)
214 
215  if isinstance(padding, tuple) and len(padding) not in [1, 2, 4]:
216  raise ValueError("Padding must be an int or a 1, 2, or 4 element tuple, not a " +
217  "{} element tuple".format(len(padding)))
218 
219  if isinstance(padding, tuple) and len(padding) == 1:
220  # Compatibility with `functional_tensor.pad`
221  padding = padding[0]
222 
223  if padding_mode not in ["constant", "edge", "reflect", "symmetric"]:
224  raise ValueError("Padding mode should be either constant, edge, reflect or symmetric")
225 
226  if padding_mode == "constant":
227  if isinstance(fill, numbers.Number):
228  fill = (fill,) * len(img.getbands())
229  if len(fill) != len(img.getbands()):
230  raise ValueError("fill should have the same number of elements "
231  "as the number of channels in the image "
232  "({}), got {} instead".format(len(img.getbands()), len(fill)))
233  if img.mode == "P":
234  palette = img.getpalette()
235  image = ImageOps.expand(img, border=padding, fill=fill)
236  image.putpalette(palette)
237  return image
238 
239  return ImageOps.expand(img, border=padding, fill=fill)
240  else:
241  if isinstance(padding, int):
242  pad_left = pad_right = pad_top = pad_bottom = padding
243  if isinstance(padding, tuple) and len(padding) == 2:
244  pad_left = pad_right = padding[0]
245  pad_top = pad_bottom = padding[1]
246  if isinstance(padding, tuple) and len(padding) == 4:
247  pad_left = padding[0]
248  pad_top = padding[1]
249  pad_right = padding[2]
250  pad_bottom = padding[3]
251 
252  if img.mode == 'P':
253  palette = img.getpalette()
254  img = np.asarray(img)
255  img = np.pad(img, ((pad_top, pad_bottom), (pad_left, pad_right)), padding_mode)
256  img = Image.fromarray(img)
257  img.putpalette(palette)
258  return img
259 
260  img = np.asarray(img)
261  # RGB image
262  if len(img.shape) == 3:
263  img = np.pad(img, ((pad_top, pad_bottom), (pad_left, pad_right), (0, 0)), padding_mode)
264  # Grayscale image
265  if len(img.shape) == 2:
266  img = np.pad(img, ((pad_top, pad_bottom), (pad_left, pad_right)), padding_mode)
267 
268  return Image.fromarray(img)

291 @torch.jit.unused
292 def resize(img, size, interpolation=Image.BILINEAR):
293  r"""Resize the input PIL Image to the given size.
294 
295  Args:
296  img (PIL Image): Image to be resized.
297  size (sequence or int): Desired output size. If size is a sequence like
298  (h, w), the output size will be matched to this. If size is an int,
299  the smaller edge of the image will be matched to this number maintaining
300  the aspect ratio. i.e, if height > width, then image will be rescaled to
301  :math:`\left(\text{size} \times \frac{\text{height}}{\text{width}}, \text{size}\right)`.
302  For compatibility reasons with ``functional_tensor.resize``, if a tuple or list of length 1 is provided,
303  it is interpreted as a single int.
304  interpolation (int, optional): Desired interpolation. Default is ``PIL.Image.BILINEAR``.
305 
306  Returns:
307  PIL Image: Resized image.
308  """
309  if not _is_pil_image(img):
310  raise TypeError('img should be PIL Image. Got {}'.format(type(img)))
311  if not (isinstance(size, int) or (isinstance(size, Sequence) and len(size) in (1, 2))):
312  raise TypeError('Got inappropriate size arg: {}'.format(size))
313 
314  if isinstance(size, int) or len(size) == 1:
315  if isinstance(size, Sequence):
316  size = size[0]
317  w, h = img.size
318  if (w <= h and w == size) or (h <= w and h == size):
319  return img
320  if w < h:
321  ow = size
322  oh = int(size * h / w)
323  return img.resize((ow, oh), interpolation)
324  else:
325  oh = size
326  ow = int(size * w / h)
327  return img.resize((ow, oh), interpolation)
328  else:
329  return img.resize(size[::-1], interpolation)
