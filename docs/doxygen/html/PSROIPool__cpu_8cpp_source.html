35  int roi_height = std::max(roi_end_h - roi_start_h, 1);

37  T bin_size_w = static_cast<T>(roi_width) / static_cast<T>(pooled_width);
38 
39  int c_in = 0;
40  for (int c_out = 0; c_out < channels_out; ++c_out) {
41  for (int ph = 0; ph < pooled_height; ++ph) {
42  for (int pw = 0; pw < pooled_width; ++pw) {
43  int hstart = static_cast<int>(floor(static_cast<T>(ph) * bin_size_h));
44  int wstart = static_cast<int>(floor(static_cast<T>(pw) * bin_size_w));
45  int hend =
46  static_cast<int>(ceil(static_cast<T>(ph + 1) * bin_size_h));
47  int wend =
48  static_cast<int>(ceil(static_cast<T>(pw + 1) * bin_size_w));
49

51  hstart = std::min(std::max(hstart + roi_start_h, 0), height - 1);
52  hend = std::min(std::max(hend + roi_start_h, 0), height - 1);
53  wstart = std::min(std::max(wstart + roi_start_w, 0), width - 1);
54  wend = std::min(std::max(wend + roi_start_w, 0), width - 1);
55  bool is_empty = (hend <= hstart) || (wend <= wstart);
56

58  input + (roi_batch_ind * channels + c_in) * height * width;
59

61  for (int h = hstart; h < hend; ++h) {
62  for (int w = wstart; w < wend; ++w) {

64  out_sum += offset_input[input_index];
65  }
66  }

157

159  at::checkAllSameType(c, {input_t, rois_t});
160

164  int width = input.size(3);
165 
166  AT_ASSERTM(
