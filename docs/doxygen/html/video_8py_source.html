91 def _read_from_stream(
92  container, start_offset, end_offset, pts_unit, stream, stream_name
93 ):
94  global _CALLED_TIMES, _GC_COLLECTION_INTERVAL
95  _CALLED_TIMES += 1
96  if _CALLED_TIMES % _GC_COLLECTION_INTERVAL == _GC_COLLECTION_INTERVAL - 1:
97  gc.collect()
98 
99  if pts_unit == "sec":
100  start_offset = int(math.floor(start_offset * (1 / stream.time_base)))
101  if end_offset != float("inf"):
102  end_offset = int(math.ceil(end_offset * (1 / stream.time_base)))
103  else:
104  warnings.warn(
105  "The pts_unit 'pts' gives wrong results and will be removed in a "
106  + "follow-up version. Please use pts_unit 'sec'."
107  )
108 
109  frames = {}
110  should_buffer = True
111  max_buffer_size = 5
112  if stream.type == "video":
113  # DivX-style packed B-frames can have out-of-order pts (2 frames in a single pkt)
114  # so need to buffer some extra frames to sort everything
115  # properly
116  extradata = stream.codec_context.extradata
117  # overly complicated way of finding if `divx_packed` is set, following
118  # https://github.com/FFmpeg/FFmpeg/commit/d5a21172283572af587b3d939eba0091484d3263
119  if extradata and b"DivX" in extradata:
120  # can't use regex directly because of some weird characters sometimes...
121  pos = extradata.find(b"DivX")
122  d = extradata[pos:]
123  o = re.search(br"DivX(\d+)Build(\d+)(\w)", d)
124  if o is None:
125  o = re.search(br"DivX(\d+)b(\d+)(\w)", d)
126  if o is not None:
127  should_buffer = o.group(3) == b"p"
128  seek_offset = start_offset
129  # some files don't seek to the right location, so better be safe here
130  seek_offset = max(seek_offset - 1, 0)
131  if should_buffer:
132  # FIXME this is kind of a hack, but we will jump to the previous keyframe
133  # so this will be safe
134  seek_offset = max(seek_offset - max_buffer_size, 0)
135  try:
136  # TODO check if stream needs to always be the video stream here or not
137  container.seek(seek_offset, any_frame=False, backward=True, stream=stream)
138  except av.AVError:
139  # TODO add some warnings in this case
140  # print("Corrupted file?", container.name)
141  return []
142  buffer_count = 0
143  try:
144  for _idx, frame in enumerate(container.decode(**stream_name)):
145  frames[frame.pts] = frame
146  if frame.pts >= end_offset:
147  if should_buffer and buffer_count < max_buffer_size:
148  buffer_count += 1
149  continue
150  break
151  except av.AVError:
152  # TODO add a warning
153  pass
154  # ensure that the results are sorted wrt the pts
155  result = [
156  frames[i] for i in sorted(frames) if start_offset <= frames[i].pts <= end_offset
157  ]
158  if len(frames) > 0 and start_offset > 0 and start_offset not in frames:
159  # if there is no frame that exactly matches the pts of start_offset
160  # add the last frame smaller than start_offset, to guarantee that
161  # we will have all the necessary data. This is most useful for audio
162  preceding_frames = [i for i in frames if i < start_offset]
163  if len(preceding_frames) > 0:
164  first_frame_pts = max(preceding_frames)
165  result.insert(0, frames[first_frame_pts])
166  return result

169 def _align_audio_frames(aframes, audio_frames, ref_start, ref_end):
170  start, end = audio_frames[0].pts, audio_frames[-1].pts
171  total_aframes = aframes.shape[1]
172  step_per_aframe = (end - start + 1) / total_aframes
173  s_idx = 0
174  e_idx = total_aframes
175  if start < ref_start:
176  s_idx = int((ref_start - start) / step_per_aframe)
177  if end > ref_end:
178  e_idx = int((ref_end - end) / step_per_aframe)
179  return aframes[:, s_idx:e_idx]
