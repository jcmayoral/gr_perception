30 def unfold(tensor, size, step, dilation=1):
31  """
32  similar to tensor.unfold, but with the dilation
33  and specialized for 1d tensors
34 
35  Returns all consecutive windows of `size` elements, with
36  `step` between windows. The distance between each element
37  in a window is given by `dilation`.
38  """
39  assert tensor.dim() == 1
40  o_stride = tensor.stride(0)
41  numel = tensor.numel()
42  new_stride = (step * o_stride, dilation * o_stride)
43  new_size = ((numel - (dilation * (size - 1) + 1)) // step + 1, size)
44  if new_size[0] < 1:
45  new_size = (0, size)
46  return torch.as_strided(tensor, new_size, new_stride)

68 class VideoClips(object):
69  """
70  Given a list of video files, computes all consecutive subvideos of size
71  `clip_length_in_frames`, where the distance between each subvideo in the
72  same video is defined by `frames_between_clips`.
73  If `frame_rate` is specified, it will also resample all the videos to have
74  the same frame rate, and the clips will refer to this frame rate.
75 
76  Creating this instance the first time is time-consuming, as it needs to
77  decode all the videos in `video_paths`. It is recommended that you
78  cache the results after instantiation of the class.
79 
80  Recreating the clips for different clip lengths is fast, and can be done
81  with the `compute_clips` method.
82 
83  Arguments:
84  video_paths (List[str]): paths to the video files
85  clip_length_in_frames (int): size of a clip in number of frames
86  frames_between_clips (int): step (in frames) between each clip
87  frame_rate (int, optional): if specified, it will resample the video
88  so that it has `frame_rate`, and then the clips will be defined
89  on the resampled video
90  num_workers (int): how many subprocesses to use for data loading.
91  0 means that the data will be loaded in the main process. (default: 0)
92  """

194  def compute_clips_for_video(video_pts, num_frames, step, fps, frame_rate):
195  if fps is None:

213  def compute_clips(self, num_frames, step, frame_rate=None):
214  """
215  Compute all consecutive sequences of clips from video_pts.
216  Always returns clips of size `num_frames`, meaning that the
217  last few frames in a video can potentially be dropped.
218 
219  Arguments:
220  num_frames (int): number of frames for the clip
221  step (int): distance between two clips
222  """
223  self.num_frames = num_frames
224  self.step = step
225  self.frame_rate = frame_rate
226  self.clips = []
227  self.resampling_idxs = []
228  for video_pts, fps in zip(self.video_pts, self.video_fps):
229  clips, idxs = self.compute_clips_for_video(
230  video_pts, num_frames, step, fps, frame_rate
231  )
232  self.clips.append(clips)
233  self.resampling_idxs.append(idxs)
234  clip_lengths = torch.as_tensor([len(v) for v in self.clips])
235  self.cumulative_sizes = clip_lengths.cumsum(0).tolist()

249  def get_clip_location(self, idx):
250  """
251  Converts a flattened representation of the indices into a video_idx, clip_idx
252  representation.
253  """
254  video_idx = bisect.bisect_right(self.cumulative_sizes, idx)
255  if video_idx == 0:

263  step = float(original_fps) / new_fps
264  if step.is_integer():
